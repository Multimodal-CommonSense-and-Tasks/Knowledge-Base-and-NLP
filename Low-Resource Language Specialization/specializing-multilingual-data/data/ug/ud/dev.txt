مەن بولسام شەكىل ئۆزگەرتىپ ئۆزۈم قەدىرلەپ سۆيگەن بىر قىزنىڭ تومۇرىدىكى ئىسسىق،
قىزىل قېنىنى شوراۋاتقان يەتتە باشلىق يالماۋۇزغا ئايلىنىپ، كىمدۇر بىرىنىڭ ئاتلىق
كېلىپ مېنى ئۆلتۈرۈۋېتىشىنى كۈتەتتىم. لېكىن مەندىن ئىبارەت بۇ يەتتە باشلىق
يالماۋۇزنى قەتلى قىلغۇدەك ئۆزۈمدىن بۆلەك يەنە بىر تىرىك جان مەۋجۇت ئەمەس ئىدى.
چۈنكى مېنىڭ مەۋجۇت رېئاللىقتىكى ھاياتىم مىسالى بىر تور بولسا، مەن بۇ تورنى
تولىمۇ ئەستايىدىل، تولىمۇ پۇختا رەۋىشتە توقۇپ چىققانىدىم. بۇ تور ئىچىدە ياشاش
مەن ئۈچۈن قانچىلىك ئازابلىق بولسا بۇ توردىن ئازاد بولۇش ئۇنىڭدىنمۇ قورقۇنچلۇق
ئىدى. ناۋادا مەن بۇ تورنى بۇزۇپ تاشلىساملا بىر پۈتۈن جەمئىيەت تەرىپىدىن خۇددى
بىر سېۋەت ئەخلەتتەك تاشلىنىپ كېتىشىم مۇقەررەر ئىدى. بۇنىڭ بىلەن سېنىڭ قەلب
قەسرەڭدىكى يارقىن ئوبرازىم كاۋا پېلىكىدەك قۇرۇپ، گۆشى يوق قۇرۇق ئىسكىلىت
جازىسىغا ئايلىنىپ قالاتتى. ئەمەلىيەتتە مېنىڭ بۇ قاباھەتلىك رېئاللىقنى كۆرگۈم يوق
ئىدى. بۇ قاباھەتلىك رېئاللىقنى قوبۇل قىلىش مەن ئۈچۈن ئۆزۈمنىڭ ئەڭ پاجىئەلىك
ھالاكىتىمنى قوبۇل قىلىشتىنمۇ مۈشكۈل ئىدى. شۇنىڭ ئۈچۈن مەن ئۆز ئىرادەمدە ئاشۇ تور
ئىچىدە داۋاملىق ياشاۋېرىشكە نىيەت قىلدىم. شۇنداق بولسىمۇ تەن چۆلۈمنىڭ يىراق
باياۋانىدىكى روھىيەت بوشلۇقۇمدا كۈچلۈك بىر قارا بوران ۋەھىمىلىك ھۇۋلايتتى. مېنىڭ
يەككە ئىگىلىكتىكى ئاڭ قاتلىمىمدا يەر شارىنىڭ تەنھالىقى، ئىنسانىيەتنىڭ غېرىبلىقى
تېنىم بىلەن قوشۇلۇپ روھىيەت بوشلۇقۇمدىكى ئاشۇ قارا بورانغا يەم بولاتتى. پەقەت
سېنىڭ بىر جۈپ قاپقارا قوي كۆزۈڭلا زۇلمەتنىڭ يىراق بىر بۇرجىكىدىكى سۈزۈك ۋە يېشىل
بىر بوشلۇقتا ماڭا قاراپ نۇرلۇق يۇلتۇزدەك چاقناپ تۇراتتى. ماڭا مەنسۇپ بولغان بۇ
تەن ۋە روھىمدىكى بەرباتلىق خۇددى بىر بۆلەك ئېقىن سۈيى بولۇپ، زۇلمەت قاراڭغۇلۇق
ئىچىدىكى قورقۇنچلۇق بوشلۇقتىن نۇر مەرۋايىتلىرى خرۇستالدەك يالتىراپ تۇرغان سۈزۈك
ۋە يېشىل بوشلۇقۇڭغا قاراپ يېقىملىق شىلدىرلاپ ئاقاتتى. ئەمەلىيەتتە سېنىڭ ئۇ سۈزۈك
ۋە يېشىل بوشلۇقۇڭغا قاراپ يېقىملىق شىلدىرلاپ ئېقىۋاتقان بۇ نەرسە سېنى ئاستا -
ئاستا نابۇت قىلىۋاتقان ھالاكەت ئىدى. مەن ئۆز رېئاللىقىمدا لالە ئىسىملىك بىر
پەرىزاتنى سۆيگىنىم بىلەن، سەن ئۆز رېئاللىقىڭدا كامال ئىسىملىك بىر ئالۋاستىنى
سۆيگەن ئىدىڭ. ھەتتا سەن بۇ ئالۋاستىنى ئۆز تەسەۋۋۇرۇڭدا پەرىشتە سۈپىتىدە ھېس
قىلىسەنۇ، لېكىن بۇ پەرىشتىنىڭ ئاغزى - بۇرنىنىڭ سېنىڭ ئىسسىق، قىزىل قان
شەربەتلىرىڭ بىلەن بويالغان قارا پەرىشتە ئىكەنلىكىنى خىيال ۋاداڭدا ياكى پەرەز
ئېكىنزارلىقىڭدا ئەسلا كۆكلەتمەيسەن. ئەمما مەن ھامان قارا قانىتىم بىلەن سېنىڭ
ھايات شاخلىرىڭدىن ئۈمىد مېۋىلىرىڭنى قاقىمەن. قارا رەڭلىك يېقىملىق ئاۋازىمدا
سېنىڭ ۋاقىت ئۆلچىمىدىن بۇرۇن گۇگۇمدەك لەيلەپ يېتىپ كېلىۋاتقان ھالاكىتىڭنى كۈيلەپ
سايرايمەن. قارا قەلبىم بىلەن سېنىڭ روھ تېمىڭغا بىر سىقىم لايدەك چاپلىشىمەن.
«ئۆلۈم» دېگەن خەت يېزىلغان بارماقلىرىم بىلەن بوستان چاچلىرىڭنى تاراپ، يۈزلىرىڭنى
سىلايمەن. ئىسسىق نەپەسلىرىم بىلەن سېنىڭ روھىي قۇۋۋىتىڭنى ھۇزۇر بىلەن شورايمەن.
ھايات توغرىقىڭنىڭ ئۆمۈر يىلتىزىنى قۇرتتەك ئاۋايلاپ يەيمەن. مېنىڭ بۇ ھەرىكەتلىرىم
تاكى سېنىڭ روھ قۇشۇڭ تەن قەسرىڭدىن پەرۋاز قىلىپ چىقىپ كەتكىچە داۋاملىشىشى
مۇمكىن. قەدىرلىك لالە، مەن مۇشۇنداق قارا قىسمەت ئىچىدە تۇرۇپ ئۆزۈمنىڭ قارا
پەرىشتىگە، ئالۋاستىغا، يالماۋۇزغا ئايلىنىپ كېتىشتىن بۇرۇنقى بۇلاق سۈيىدەك پاك ۋە
سۈزۈك دەۋرىمنى ئەسلەيمەن. ئۇ چاغلاردا ھەر ئىككىمىز قەلبى گۈزەل ئارزۇ -
ئارمانلار، شېرىن ئەسلىمە گۈللىرى بىلەن كۆركەم بېزەلگەن 18 ياشلاردىكى بالىلار
ئىدۇق. بۇرۇتلارنىڭ خەت تارتىپ قېلىشى، قىزلىق زىنىخلارنىڭ پەيدا بولۇپ قېلىشى
بىزنىڭ بىر ھالەتتىن يەنە بىر ھالەتكە ئۆتۈۋاتقانلىقىمىزدىن ئۈنسىز بېشارەت
بېرەتتى. ھەر ئىككىلىمىزنىڭ ئۆيى شەھەردىن يىراقتىكى ناھىيىدە بولۇپ، ھەر ئىككىمىز
ناھىيە مەركىزىدىكى توپا تاملىق كونا مەكتەپنىڭ يۈزلىرى يېرىق، ئادەم ئولتۇرۇپ
بولغىچە مىدىرلاپ يېقىمسىز ئاۋازدا غىچىرلاپ كېتىدىغان، جىگەررەڭ سىرلىرى ئۆڭۈپ
كەتكەن ياغاچ پارتا - ئورۇندۇقلۇق سىنىپىدىكى ساۋاقداشلاردىن ئىدۇق. ئۆيىمىز بولسا
ئارا تام قوشنا ئىدى. مەن ئېسىمنى بىلسەم سەن بىلەن بىللە ئويناپ چوڭ
بولۇۋاتقانىكەنمەن. سەن شۇنداق ياۋاش قىز ئىدىڭ. ئۆيۈڭنىڭ بارلىق ئىشلىرىنى ھاردىم
- تالدىم دېمەي قىلاتتىڭ، سەھەرلەردە مەن بىلەن بىللە ماللارغا ئوت ئەكېلەتتىڭ.
شۇنداقتىمۇ قاپىقىدىن مۇز يېغىپ تۇرىدىغان سۆرۈن تەلەت داداڭنىڭ ساڭا بۇزۇق
كانايدەك كاركىراپ ۋارقىراشلىرى، سېنى شۇمدىن شۇمغا سېلىپ تىللاپ ئۇرۇشلىرىنى پات -
پات ئاڭلاپ، كۆرۈپ قالاتتىم. بۇنداق چاغلاردا سەن مەھەللە كوچىسىنىڭ ئارقىسىدىكى
ئەمدىلەتىن بالا بويى بولۇپ ئۆسكەن قويۇق، دىمىق قوناقلىق بويىدا ئۇزاقتىن - ئۇزاق
مۇڭلىنىپ ئولتۇرۇپ كېتەتتىڭ. گاھىدا يۈز قىسمىڭدا، گاھىدا بويۇن ياكى بىلەكلىرىڭدە
پەيدا بولۇپ قالىدىغان توق قىزىل رەڭلىك تاياق ئىزلىرى ماڭا خۇددى سېنى چېقىۋاتقان
زەھەرلىك يىلاندەك، ياكى قان ئىچىدە سوزۇلۇپ ياتقان سۆرۈن تەلەت داداڭنىڭ غەزەپتىن
قىزىرىپ كەتكەن ئىككى كۆزىدەك كۆرۈنۈپ كېتەتتى. بالىلىق خىياللىرىمدا بولسا ئۇ
زەھەرلىك يىلاننى كۆك چىۋىق بىلەن ئۇرۇپ ئۆلتۈرۈۋەتكۈم، داداڭنى قانداقتۇر بىر
سېھىر - ئەپسۇن بىلەن ساڭا زۇلۇم سالالماس قىلىپ قويغۇم كېلىپ كېتەتتى. راستىنى
دېسەم، ساڭا بەكمۇ ئىچىم ئاغرىيتتى. تاياق ئىزى قېتىپ قالغان يۈز - بەدەن، بويۇن -
بىلەكلىرىڭدىكى توق قىزىل رەڭلىك جاراھەتلەرگە قاراپ كىچىككىنە تېنىڭدىكى بىر
ئوتلام تاتلىق جېنىڭنىڭ قانچىلىك بىئارام بولۇۋاتقانلىقىنى غۇۋا تەسەۋۋۇر قىلاتتىم.
سەن بولساڭ خۇددى روھىڭ تېنىڭدىن ئۇچۇپ چىقىپ كەتكەندەك ئۈنسىز ئولتۇراتتىڭ. سېنى
يەنە ئۇردىمۇ، لالە؟ يېنىڭغا كېلىپ بوش، ئەمما سەمىمىي ئاۋازدا سەندىن سورايتتىم.
ئۇ چاغلاردا جاراڭلاپ چىقىدىغان ئاۋازىمنىڭ نېمىشقا بوش چېقىپ قالىدىغانلىقىنى
ئەسلا بىلمەيتتىم. سەن ئىككى تىزىڭنى قۇچاقلاپ تۇرغان بىلەكلىرىڭدە ماغدۇرسىز
ياتقان بېشىڭنى ئاستا كۆتۈرۈپ، سوئالىمغا جاۋاب بەرمەستىن سۇسلا كۈلۈمسىرەپ
قوياتتىڭ. كۆزۈم لېۋىڭدىكى ئاشۇ سۇس كۈلكە بىلەن قوشۇپ ئىككى كۆزۈڭنىڭ سۈزۈك
كۆكسىدە لىغىلداپ تۇرغان ياش تۇمانلىرىنى بىرگە كۆرۈپ ئۈلگۈرەتتى. سەن مېنىڭ
سورىغان سوئالىمغا ئاشۇنداق سۇس كۈلكە بىلەن جاۋاب بېرىپ بولغاندىن كېيىن بېشىڭنى
يەنە قايتا ئىككى تىزىڭنى قۇچاقلاپ تۇرغان بىلىكىڭگە ئاستا قويۇپ، ئۇزاقتىن - ئۇزاق
سۈكۈتكە چۆمۈپ نېمىلەرنىدۇر خىيال قىلىپ كېتەتتىڭ. بۇ چاغدىكى سەن خۇددى بىر پارچە
چىرايلىق تاشقا ئايلىنىسەنكى، بالا بويى ئۆسكەن قوناق شاخلىرىنىڭ شىلدىرلىغان
ئاۋازىمۇ، تومۇزغا - چېكەتكىلەرنىڭ بىر - بىرىگە ئۇلاپ يېقىمسىز سايراشلىرىمۇ سېنىڭ
ئۇزاقتىن - ئۇزاققا سوزۇلىدىغان سۈكۈتۈڭگە دەخلى قىلالمايتتى. مەن بولسام
ئالدىمدىكى سەندىن ئىبارەت بۇ كىچىككىنە چىرايلىق تاشقا قاراپ ئىچ ئاغرىقى تارتقان
ھالدا كۆزلىرىمنى چىمىلداتقىنىمچە تۇرۇپ قالاتتىم. سەن بولساڭ يېتىم قوزىدەك بويۇن
قىسىپ ئولتۇراتتىڭ. خۇددى بىز ئوقۇغان چۆچەكلەر ئىچىدىكى يەنە بىر چۆچەكتە يىرتقۇچ
بۆرە ئوماق، گۇناھسىز قوزىچاقنى يېيىش مەقسىتىگە يېتىش ئۈچۈن تۇتامى يوق سەۋەبلەرنى
ئاق يەردىنلا تاپقاندەك، سۆرۈن تەلەت داداڭمۇ ساڭا زۇلۇم قىلىش ئۈچۈن تۈكى يوق
سەۋەبلەرنى تېپىشقا ھەر ۋاقىت تەييار ئىدى. مەن بالىلىق خىياللىرىمدا ئۇ سۆرۈن
تەلەت داداڭنى ساڭا زۇلۇم قىلمىسا خاتىرجەم يۈرەلمەمدىغاندۇر، دەپمۇ ئويلاپ
قالاتتىم. لېكىن سەن كۆزلىرىڭنىڭ سۈزۈك كۆكسىدە لىغىلداپ تۇرغان ياشلارنى زادىلا
كۆز قارىچۇقۇڭنىڭ سىرتىغا چىقارمايتتىڭ. ئەكسىچە ئۇ مەرۋايىتتەك سۈزۈك ياش
تامچىلىرىنى كۆز زېمىنىڭدىكى مۇنبەت تۇپراق باغرىغا سىڭدۈرۈپ تاشلايتتىڭ. دەۋرەپ
تۇرغان كۆز ياشلىرىڭغا تۇغان سېلىش سېنىڭ ئىلكىڭدىكى ئەڭ گۈزەل سەنئەت ئىدى. مەن
كۆپ ھاللاردا چوڭلار بالىلارنى نېمىشقا ئۇرىدىغاندۇر دەپ ئويلايتتىم. لېكىن
بالىلىقتىكى گۆدەك، ساددا تەپەككۇرۇم بىلەن ئۆزۈمنىڭ سوئالىغا ئۆزۈم جاۋاب
تاپالمايتتىم. كىچىككىنە مېڭەم بولسا مانا مۇشۇنداق جاۋابى يوق سوئاللار بىلەن
ئاشلىق بېسىلغان ئامباردەك توشۇپ كېتەتتى. − − كامال، سەن تاياق يەپ باققانمۇ؟ سەن
تىزىڭنى مەھكەم قۇچاقلاپ تۇرغان يۇمشاق ۋە ئاق بىلەكلىرىڭ ئۈستىدە ماغدۇرسىز
ھالەتتە ياتقان بېشىڭنى كۆتۈرمەيلا سوراپ قالدىڭ. ئاۋازىڭ ئۇزاق يىل ئېغىر كېسەل
تارتىپ ماغدۇرى قالمىغان ئادەمنىڭ ئاۋازىدەك تىترەڭگۈ ۋە جانسىز بولۇپ، ھەسرەت
تۇمانلىرى لەيلەپ تۇرغان بىر جۈپ قاپقارا قوي كۆزلىرىڭ نامەلۇم بىر خىيال شولىلىرى
ئىچىدە نەگىدۇر بىر يەرگە ئىپادىسىزلا تىكىلىپ قالغانىدى. سېنىڭ تولىمۇ ئاددىي،
بىراق بالىلىق جاراھەت ئىزلىرى بىلەن كۆل سۈيىدەك داۋالغۇپ تۇرغان قەلب قەسرەڭدىن
كۈزدىكى دەرەخ شېخىدىن ئۈزۈپ چۈشۈۋاتقان سارغۇچ ياپراقتەك لەيلەپ چىققان سوئالىڭ
گوياكى يېلىنجاپ كۆيۈۋاتقان بىر پارچە قىزىل چوغدەك يۈرەك - باغرىمنى ۋاژىلدىتىپ
كۆيدۈرۈپ، ئاۋازىڭ قۇلاق تۈۋۈمدە دەھشەتلىك گۈلدۈرمامىدەك گۈلدۈرلەپ ئۆتتى. كاللام
بولسا ھاياتلىق كۆرۈنۈشى مەۋجۇت بولمىغان ئۆلۈك چۆللۈكتەك قۇپقۇرۇق ھەم جىمجىت
ئىدى. بۇ قۇپقۇرۇق ئۆلۈك چۆللۈكتە ھەتتا بىر تىنىق ھاۋامۇ مەۋجۇت ئەمەس ئىدى. لېكىن
بۇ ھاياتلىق كۆرۈنۈشى مەۋجۇت بولمىغان چۆللۈكنىڭ چەكسىز كەتكەن تۆت بۇلۇڭىدىن سېنىڭ
ئاۋازىڭ خۇددى غايىبتىن كېلىۋاتقاندەك ئۈزۈلمەي ياڭراپ تۇراتتى. مەن بولسام بۇ
ھەسرەتلىك ئاۋازنىڭ تاۋۇش مەركىزىدە كۆز يەتكۈسىز يىراق كۆككە قاراپ خۇددى شامال
چاقپەلىكىدەك توختاۋسىز پىرقىرايتتىم. پۈتۈن چۆل، پۈتۈن ئاسمان باش ئۈستۈمدە نۇر
چېچىپ يېلىنجاپ تۇرغان قۇياش بىلەن قوشۇلۇپ چۆرگىلەيتتى. مەن ئىچىمنىڭ قىزىپ،
كالپۇكلىرىمنىڭ قۇرۇپ گەز باغلاۋاتقانلىقىنى، ئۇچۇق چىققاندەك تىكەنلىشىپ كەتكەن

ئەۋجىدە ۋۇجۇدۇڭ چايقىلىپ، روھىڭ يايراپ ئۆزۈڭنى ئانا تەبىئەتنىڭ بىر قىسمىدەك ھېس
قىلىسەن. ئاشۇ ئۈزۈلمەس ناخشىلار بىلەن ئايىغى چىقماس پاراڭلار سېنىڭ بالىلىق
قەلبىڭنى ئۆتمۈش يىللارنىڭ گىرىمسەن قوينىغا باشلاپ كېتىدۇ. ئۆزۈڭمۇ تۇيمىغان ھالدا
ئۇزاقتىن ئۇزاق خىياللارغا غەرق بولىسەن. تۇيۇقسىز ئۈستۈڭدىكى پايانسىز ئاسماندا
غۇژمەكغۇژمەك بولۇشۇپ، تىنىمسىز جىمىرلاۋاتقان يورۇق يۇلتۇزلاردىن بىرىنىڭ ساقىپ
چۈشكەنلىكىنى كۆرىسەندە، ئاندىن ئېسىڭگە كېلىسەن. ئەنە شۇ چاغدا گويا سەھەر
ئاسمىنىغا چېكىلگەن نەقىشلەردەك كۆرۈنىدىغان ساناقسىز يۇلتۇزلار بىلەن ئالەمنىڭ ئۇ
چېتىدىن بۇ چېتگە سوزۇلغان ھالدا ئاقىرىپ تۇرغان سامان يولى دىققىتىڭنى تارتىدۇ.
پاھ، ئاسمان بوشلۇقىدا يۇلتۇزلار نېمە دېگەن جىق – ھە! ئۇنى ساناپمۇ، قاراپمۇ
تۈگىتەلمەيسەن. بۇ يۇلتۇزلار ساڭا شۇنداق يېقىن يەردىلا چاقناۋاتقاندەك، قولۇڭنى
سوزىساڭلا يېتىدىغاندەك تۇيۇلىدۇ. بۇ ساناقسىز يۇلتۇزلارنىڭ بەزىسى شوخ قىزلارنىڭ
كۆزلىرىدەك نۇرسىز يالتىرايدۇ. تىنىمسىز جىمىرلاپ تۇرغان يۇلتۇزلار قەلبىڭدە
يېڭىدىن يېڭى بالىلىق سوئاللارنى ئويغىتىدۇ؛ - بۈگۈن ئاخشام قايسى ھۆر قىزلارنىڭ
بوينىدىكى ئۈنچە – مەرۋايىتلىرى چېچىلىپ كەتكەندۇ؟ قايسى چۆچەكچى موماينىڭ ھېكايە –
چۆچەك ساندۇقىنىڭ ئاغزى ئېچىلىپ كەتكەندۇ؟ - ئۇ ئاتا – بوۋىلىرىمىزنىڭ چاقناپ
تۇرغان ئەقىل چىرىغىمىدۇ؟ ياكى بىتەلەي بالىلارنىڭ قۇت نىشانىمىدۇ؟ تۆتكۆز بىلەن
كۈتەكۈتە، شەرقشامىلى سوقۇپ، باھارنىڭ قەدىمىمۇ يېقىنلاشتى. بارلىق مەۋجۇدات، گويا
ئەمدىلا ئۇيقۇدىن ئويغانغاندەك، لاپلا قىلىپ كۆزلىرىنى ئاچتى. قارلار يالتىرىغىلى،
سۇلار ئۇلغايغىلى، قۇياش قىزارغىلى باشلىدى. يۇمران، ياپيېشىل گىياھلار توپا
ئاستىدىن بىلىندۈرمەي باش كۆتۈردى، ھويلا - ئارام، ئېتىز - دالىلارغا نەزەر
سالسىڭىز، ھەممە يەر يېشىلچىلىققا تولدى. ئۇنىڭغا قاراپ بىردەم ئولتۇرۋالغىڭىز، بىر
پەس مۆكۈ - مۆكۈلەڭ ئوينىۋالغىڭىز كېلەتتى. شامال لەرزان سوقاتتى، ئوتتچۆپلەر
كىمخابتەك يۇمشاق ئىدى. شاپتۇل، ئۆرۈك، نەشپۈت دەرەخلىرى بەسبەستە ھۈپپىدە چېچەك
ئاچتى. چېچەكلەرنىڭ قىزىلى چوغدەك، ھالى شەپەقتەك، ئاقلىرى بەئەينى قاردەك ئىدى.
چېچەكلەردىن خۇشبۇي پۇراقلار گۈپۈلدەپ تۇراتتى. قۇلۇلىنڭ ئۇچقۇدەك ھالى قالماپتۇ.
قېنى، ئېيتىڭلارچۇ، بۇ نېمە ئۈچۈن؟ - چۈنكى ئالتۇن بېلىقنڭ قاسىرىقى بار، _ دەپتۇ
بىر بالا. چوقۇم ئالتۇن بېلىق سۇنى ئىچىۋالغان، _ دەپتۇ يەنە بىرەيلەن. ئىلىنا
ئۇلارنىڭ دېگەنلىرىنى توغرا تاپماپتۇ، ئەمما ئۆزىمۇ بۇنىڭ سەۋەبىنى ئويلاپ
تاپالماپتۇ. ئۇ ئۆيگە بارغاندا، ئالىمنىڭ دېگەنلىرىنى ئانىسىغا سۆزلەپ بەرگەنىكەن،
ئانىسى دەپتۇ: - ئويلاۋەرگىنىڭ بىلەن بىكار، ئۆزۈڭ سىناپ كۆرۈپ باقمامسەن! ئىلىنا
بىر شېشىگە سۇ توشقۇزۇپ، بىر ئالتۇن بېلىقنى ساپتۇ. بېلىق شېشىگە سېلىنغان ھامان سۇ
تېشىپ كىتىپتۇ. بۇنى كۆرگەن ئىلىنا چۆچۈپ كېتىپتۇ: ۋاي، بۇ نېمە بولغىنى؟
لونزىۋېننىڭ دېگىنى بىلەن ئوخشاش چىقمىدىغۇ؟ ئويلىغانسېرى ئىلىنانىڭ ئاچچىقى كەپتۇ.
ئەتىسى ئەتىگەندە، ئۇ لونزۋېننىڭكىگە كېلىپ دەپتۇ: - بىزدىن ئۇنداق سوئالنى
سورىغىنىڭىز نېمىسى؟ بىزنى كىچىك بالا دەپ، بىر كولدۇرلىتاي دېگەنمىدىڭىز يا؟
لونزىۋېن قاقاقلاپ كۈلۈپ كېتىپتۇ ۋە مۇنداق دەپتۇ: - سىلەرنى كولدۇرلىتىش ئويۇم
يوق. مەن سىلەرگە ئالىملارنىڭ دېگەنلىرىنىڭ ھەممىسىلا توغرا بولۇپ كېتىشى ناتايىن،
قول سېلىپ ئىشلەپ، سىناپ كۆرۈش كېرەك، دېگەننى بىلدۈرۈپ قويماقچىمەن. لونزۋېننىڭ
سۆزىنى ئاڭلاپ، ئىلىنانىڭ چىرايىغا كۈلكە يۈگۈرۈپتۇ. بىز ئۆگىنىش ۋە تۇرمۇشتا قول
سېلىپ ئىشلىمىسەك ۋە مېڭە ئىشلاتمىسەك بولمايدۇ. ھەربىر بايقاش ۋە ھەربىر كەشپىيات
كىشىلەرنىڭ كۆڭۇل قويۇپ پىكىر يۈرگۈزۈشىدىن بارلىققا كەلگەن. بىزمۇ ئىجتىھاتلىق
ئادەم بولۇپ، ئۆز ئەجرىمىز ۋە ئەقىل پاراسىتىمىز بىلەن يېڭىيېڭى نەرسىلەرنى
بايقايلى، ئىجاد قىلايلى! 20ەسىرنىڭ 20يىللىرىنڭ مەلۇم بىر كۈنى، مەمتىلى ئىسىملىك
بىر يىگىت بۈيۈك مەقسەتلەرنى كۆڭلىگە پۈكۈپ قەشقەردىن ئۇزاق سەپەرگە ئاتلىنىپتۇ. ئۇ
چاغلاردا بۇ زېمىندا ماشىنا، پويىزلار بولمىغاچقا، ئات، ئېشەك، تۆگىلەرگە مىنىپ
سەپەر قىلىشقا توغرا كېلەتتى. لېكىن بۇ جاپالىق ئۇزۇن سەپەر بۇ يىگىتنى بىلىم
ئىزدەش نىيىتىدىن ياندۇرالماپتۇ. ئۇ تالاي مۇشەققەتلەرنى باشتىن كەچۈرۈپ، نەچچە ئاي
يول يۈرۈپ، ئاخىرى تۈركىيىگە يېتىپ بېرىپتۇ. مەمتىلىنڭ يانچۇقى قۇرۇق ئىكەن، يەنە
كېلىپ تۈركىيىدە تونۇشبىلىشلىرىمۇ يوق ئىكەن. ئۇ بوينىغا تەنزە ئېسىپ، ئۇششاقچۈششەك
نەرسىلەرنى سېتىپ جان بېقىپتۇ، ھەر ۋاقىت بىلىم ئېلىش پۇرسىتىنى ئىزدەپتۇ. ئۇزاق
ئۆتمەي ئۇ بىر ئوقۇتقۇچى يېتىشتۈرۈش مەكتىپىدە قارا خىزمەتچى بوپتۇ. مەمتىلى ھەر
كۈنى مەكتەپنىڭ تازىلىق ئىشلىرىنى ئىشلەپ بولۇپ، سىنىپ دېرىزىسىنڭ سىرتىدا تۇرۇپ،
پۈتۈن دىققىتى بىلەن دەرس ئاڭلايدىغان بوپتۇ. بىر كۈنى، مەكتەپ مۇدىرى مەمتىلىنى
ئىشخانىسىغا چاقىرىتىپ بۇ ئىشنى سۈرۈشتۈرۈپتۇ. _ كەچۈرۈڭ، ئەپەندى مەن ئىشلىرىمنى
تۈگىتىپ بولۇپ ئازراق دەرس ئاڭلاي دېۋىدىم... _ دەپتۇ مەمتىلى. _ دەرس ئاڭلاي
دېۋىدىم؟ _ مۇدىر ھەيرانلىق نەزىرى بىلەن مەمتىلىگە قاراپتۇ. _ شۇنداق، مەن بىلىم
ئېلىش كويىدا بۇ ئەلگە كەلگەن... مەمتىلى مۇدىرغا كەچۈرمىشلىرىنى سۆزلەپ بېرىپتۇ.
قاتتىق تەسىرلەنگەن مۇدىر ئۇنىڭ دولىسىغا ئۇرۇپ تۇرۇپ دەپتۇ: _ سېنىڭدەك كىشىلەر
دەرسنى دېرىزە سىرتىدا ئەمەس، دېرىزە ئىچىدە ئاڭلىشى كېرەك ئۇنى بىر سىنىپقا باشلاپ
كىرىپ ئورۇنلاشتۇرۇپ قويۇپتۇ... مەمتىلى ۋاقىتنى قەدىرلەپ تىرىشىپ ئوقۇپتۇ. ئوقۇشنى
ئەلا نەتىجە بىلەن پۈتكۈزۈپ، 1934يىلى ئەتىيازدا ئانا يۇرتىغا قايتىپ كەپتۇ. ئۇ
يۇرتداشلىرىنى سەپەرۋەر قىلىپ يېڭىچە مەكتەپلەرنى ئېچىپ، مىڭلىغان بالىلارنى بىلىم
ئېلىش پۇرسىتىگە ئېرىشتۈرۈپتۇ، كۆپلىگەن شئېرمۇزىكىلارنى ئىجاد قىلىپ، ئۇلارغا
ئۆگىتىپتۇ... بۇ كىشى بىزنىڭ مەشھۇر مائارىپچىمىز، ئوتيۈرەك كومپوزىتورىمىز ۋە
شائىرىمىز مەمتىلىئەپەندى (تەۋپىق) ئىدى. 160نەچچە يىل ئىلگىرى، فرانسىيىدە ئامپېر
ئىسىملىك داڭلىق فىزىكا ئالىمى ئۆتكەن. ئۇ بېرىلىپ ئىلمىي تەتقىقات بىلەن
شۇغۇللىناتتى. بىر كۈنى، ئامپېر كوچىدا سەيلە قىلىۋاتاتتى. يولۇچىلار، پەيتۇنلار
ئۇياقۇياققا ئۆتۈپ تۇراتتى. بىراق، ئامپېر بۇلارنى كۆرمىگەن، ئاڭلىمىغاندەك، بېشىنى
تۆۋەن سالغىنىچە كېتىۋاتاتتى. ئەسلىدە ئۇ بىر ماتېماتىكىلىق مەسلە ئۈستىدە پىكىر
يۈرگۈزۈۋاتاتتى. باشتا ئۇ كۆڭلىدە ھېسابلاپ، بارمىقى بىلەن چاپىنىنىڭ يېڭىگە
نېمىلەرنىدۇر سىزىپ ماڭدى، كېيىن بىر يەرنى تېپىپ ھېسابلىسا ياخشى بولىدىغانلىقىنى
ھېس قىلدى. سۆزلەپ كەلسەك بەك قىزىق، ئامپېر بىردىنلا كوچىنىڭ ياقىسىدا تۇرغان بىر
دوسكىنى كۆرۈپ قالدى. گويا بۇ ئۇنىڭ ئۈچۈن راسلاپ قويۇلغاندەك قىلاتتى، بەك ياخشى
بولدى! ئامپېر خۇش بولۇپ ئۇياققا باردى – دە، يانچۇقىدىن بورنى چىقىرىپ، دوسكىغا
يېزىپ ھېسابلاشقا باشلىدى. ئامپېر تازا ھېسابلاۋاتقاندا، "دوسكا" مىدىرلىدى – دە،
ئاستا ئالغا قاراپ سۈرۈلۈشكە باشلىدى. - مىدىرلىما، مىدىرلىما دەيمەن! – دىدى
ئامپېر ئالدىراپ، - يەنە بىر ئاز چىدىساڭلا ھېسابلاپ بولىمەن! لېكىن، "دوسكا" ئالغا
قاراپ كېتىۋەردى، ئامپېرمۇ ئىختىيارسىز ئۇنىڭغا ئەگىشىپ مېڭىپ، پۈتۈن زېھنى بىلەن
ھېسابلاۋەردى. كېيىن، ھېلىقى دوسكىنڭ مېڭىشى تېزلاپ كېتىپ، ئامپېر ئۇنىڭغا
يېتىشەلمەي قالدى. ئۇ بېشىنى كۆتۈرۈپ شۇنداق قارىسا، بايا ھېساب يازغىنى دوسكا
ئەمەس، بەلكى پەيتۇننىڭ كەينى تەرىپى ئىكەن.... ئالىملارنىڭ تەسىرلىك ئىشئىزلىرى
بىزگە شۇنى چۈشەندۈرىدۇكى، تىنماي ئۆگىنىش ئادەمنى چوڭچوڭ ئىشلارنى ۋۈجۇدقا چىقىرىش
ئىمكانىيىتىگە ئىگە قىلىدۇ. ئادەم ئۆگنىش بىلەن ئالغا باسىدۇ. ئەتىكى ئىشلار ئۈچۈن
بۈگۈن ئۆگىنىش كېرەك، كەلگۈسى ئۈچۈن، ھارمايتالماي ئۈزلۈكسىز ئۆگىنىش كېرەك.
ئەنقانىڭ ئۇۋا ياسىشىنى بىلىدىغانلىقىنى ئاڭلىغان بىر مۇنچە قۇشلار ئۇنىڭدىن
ماھارەت ئۆگەنگىلى بېرپتۇ. ئەنقا: - بىرەر ماھارەتنى ئۆگىنىش ئۈچۈن سەۋرچان بولۇش
كېرەك، ئۇنداق بولمىغاندا ھېچنېمىنى ئۆگەنگىلى بولمايدۇ...، - دەپ سۆز باشلىشىغا،
توخۇ ئۇخلاپ قاپتۇ. ھۇۋقۇش: "ئەنقانىمۇ كۆرۈپ قويدۇم، قۇرۇق گەپنى تولا قىلىدىكەن"
دەپ ئويلاپتۇ - دە، ئۇچۇپ كېتىپتۇ. - ياخشى ئۇۋا ياساش ئۈچۈن ئورۇننى ياخشى
تاللىشىمىز، يەنى چوڭ دەرەخ ئاچىمىقىنى... بۇ گەپنى ئاڭلىغان سار: "ھەي،
ئەسلىدەشۇنچىلىكلا بىر ئىش ئىكەنغۇ، بىلىۋالدىم!" دەپتۇ - دە، قاناتلىرىنى قېقىپ
ئۇچۇپ كېتىپتۇ. - ئېغىزدا چىشلەپ كەلگەن چاۋارلارنى بىربىرلەپ تىزىپ... سۆز شۇ
يەرگە كەلگەندە، قاغا: "ۋايۋۇي، چاۋارلارنى تىزسىلا بولىدىكەنغۇ، مەنمۇ
ئۆگىنىۋالدىم" دەپتۇ - دە، قاقىلداپ ئۇچۇپ كېتىپتۇ. - ياسىغان ئۇۋىمىز بىخەتەر
بولۇشى كېرەك. خاتىرجەم تۇراي دېسەڭ، ئۇۋاڭنى ئۆيلەرنىڭ پېشايۋانلىرىغا ياسا، ئۇ
چاغدا بوران - يامغۇردىنمۇ قورقمايدىغان بولىسەن... بۇنى ئاڭلىغان قۇشقاچ:
"ئويلىغىنىم بىلەن ئوخشاشكەن" دەپ، ئۇچۇپ كېتىپتۇ. پەقەت كىچىك قارلىغاچلا ئەنقانىڭ
سۆزىنى ئەستايىدىل ئاڭلاپ ئولتۇرۇپتۇ. ئەنقا سۆزىنى داۋاملاشتۇرۇپ: _ ئاندىن كېيىن
لاي ئەكېلىپ، لاينىڭ ھەر بىر قەۋىتىگە قىل ۋە ئوت - خەسلەرنى ياتقۇز، شۇندىلا پۇختا
ئۇۋا ياسىغان بولىسەن، - دەپتۇ. ئەنقا سۆزلەۋاتقاندا ئۇخلاپ قېلىپ ھېچنېمىنى
ئاڭلىمىغان توخۇ ئۇۋا ياساشنى بىلەلمەپتۇ، شۇڭا تا ھازىرغىچە كىشىلەر ياساپ بەرگەن
تەييار ئۇۋىدا ياشايدىكەن. ھۇۋقۇش ئۆزىنى چوڭ تۇتۇپ، ئۇۋا ياساشنى ئۆگەنمىگەچكە،
پەقەت دەرەخ شېخىدا دۈگدىيىپ ئولتۇرۇشنىلا بىلىدىكەن. سار ئاچىماق شاخنى كۆرسىلا
قونۇۋالىدىكەن. قاغا ئۇۋا ياساشنى چالا ئۆگەنگەچكە، سالغان ئۇۋىسى شامال چىقسىلا
چۇۋۇلۇپ كېتىدىكەن. قۇشقاچ بولسا، پىشايۋان ئاستى ۋە قاراڭغۇ كامارلارغا ئۇۋا
سالىدىكەن. قارلىغاچ كۆڭۈل قويۇپ ئۆگەنگەچكە، ناھايىتى ئەپچىل ۋە پۇختا ئۇۋا
ياسايدىكەن. شۇڭا، ئۇ بوران - يامغۇرلۇق چاغلاردىمۇ ئۇۋىسىدا خاتىرجەم
يېتىۋېرىدىكەن. بۇرۇننىڭ بۇرۇنىسىدا، قۇلۇلىنىڭ قىياپىتى ھازىرقىغا ئوخشىمايدىكەن.
ئۇنىڭ بىر جۈپ كۈچلۈك قانىتى بار ئىكەن، كۆكتە ئەركىن پەرۋاز قىلالايدىكەن. بىر
قېتىم ھاشاراتلار ئارا ئۇچۇش مۇسابقسى ئۆتكۈزۈلۈپتۇ. قۇلۇلە يىڭناغۇچ، ھەسەلھەرىسى،
كېپىنەك... قاتارلىق رىقابەتچىلىرىنى ئارقىدا قالدۇرۇپ، چېمپىيونلۇق لوڭقىسىنى
ئاپتۇ. شۇنىڭدىن باشلاپ، قۇلۇلە قاتتق مەغرۇرلىنىپ كېتىپتۇ. ئۇ ئۆزىنىڭ ئۇچۇش
چېمپىيونى ئىكەنلكىنى باشقىلار بىلمەي قالمىسۇن، دەپ ئەنسىرەپ، كۈن بويى مۇكاپات
لوڭقىسىنى ئۈستىدىن چۈشۈرمەپتۇ. كەچ كىرگەندە بولسا، باشقىلار ئوغرىلىۋالمىسۇن، دەپ
قورقۇپ، لوڭقىنىڭ ئىچىدە ئۇخلايدىغان بوپتۇ. مۇكاپات لوڭقىسى بەك چوڭ، بەك ئېغىر
ئىكەن. قۇلۇلە ئۇنى يۈدۈۋالغاچقا ئۇچقۇدەك ھالى قالماپتۇ. چوڭ ئوغۇل سېۋەتنى
كۆتۈرۈپ ئېرىق بويىغا چىقىپتۇ، لىكىن ھەر قانچە قىلىپمۇ سېۋەتتە سۇ توختىتالماي،
ئۆيگە قايتىپ كىرىپتۇ. بوۋاي كىچىك ئوغلىنى بۇيرۇپتۇ. بۇ بالا ئېرىق بويىغا
چىقىپتۇ. ئۇ ئۆيدىن ئېلىپ چىققان ساماننى سېغىزغا ئارلاشتۇرۇپ لاي ئېتىپتۇ ۋە لاي
بىلەن سېۋەتنى سۇۋاپتۇ. ئۇ لىق سۇ توشقۇزۇلغان سېۋەتنى كۆتۈرۈپ ئۆيگە كىرگەنىكەن،
دادىسى ئۇنىڭ پەمپاراسىتىگە ئاپىرىن ئوقۇپتۇ، ئاندىن دەپتۇ: - بالىلىرىم، مەن ئۆلەر
ھالەتتە ياتىمەن، مېنىڭ ۋەسىيىتىم شۇكى، ئەقىل ئىشلىتىشكە ماھىر بولۇڭلار، سەۋەب
قىلساڭلار سېۋەتتە سۇ توختايدۇ، ئۇزاق ئۆتمەي بوۋاي ئالەمدىن ئۆتۈپتۇ. ئىككى ئوغۇل
دادىسىنىڭ ۋەسىيىتىنى ئېسىدە چىڭ تۇتۇپ، ھەر قانداق ئىشتا پەم ئىشلىتىدىغان بوپتۇ.
شۇڭا، ئۇلار ئۆمرىنىڭ ئاخىرىغىچە بەخىتلىك ياشاپتۇ. «سەۋەب قىلساڭ سېۋەتتە سۇ
توختايدۇ» دېگەن ماقال ئەنە شۇنىڭدىن قالغانىكەن. - مۈشۈكتىن قانداق قۇتۇلارمىز؟ -
دەپ مەسلىھەتلىشىپتۇ چاشقانلار. - مۈشۈك دېگەن بەك مەككار، ئۇنىڭدىن قۇتۇلغىلى
بولماسمىكىن، - دەپتۇ بىر چاشقان. - ئۇ ماڭغان چاغدا شەپە چىقارمايدۇ ئەمەسمۇ، -
دەپتۇ يەنە بىر چاشقان ئۇنىڭ گېپىنى ماقۇللاپ، - يەنە كېلىپ، ئۇ مۈكۈنۈپ تۇرسا
بىلەلمەي قالىدىكەنمىز. - مەن بىر ئامال تاپتىم! - دەپتۇ بىر ۋىجىك چاشقان، -
مۈشۈكنىڭ بوينىغا قوڭغۇراق ئېسىپ قويساق، ئۇ بىزگە يېقىنلاشماي تۇرۇپلا قوڭغۇراقنىڭ
ئاۋازىنى ئاڭلاپ قالىمىز - دە، بەدەر تىكىۋېتىمىز. بۇ چارە چاشقانلارغا يېقىپ
قاپتۇ، ئۇلار شۇ بويىچە ئىش كۆرۈشنى قارار قىپتۇ. لېكىن، ئەڭ قېرى بىر چاشقان
ئۇلاردىن سوراپتۇ: - بۇغۇ ئوبدان ئامال ئىكەن، ئەمما مۈشۈكنىڭ بوينىغا قوڭغۇراقنى
كىم ئاسىدۇ؟ چاشقانلار بۇ سۇئالغا نېمىدەپ جاۋاب بېرىشىنى بىلەلمەي بىربىرىگە
قارىشىپتۇ، ئارقىدىن «سەن ئاس» دېسە، «سەن ئاس» دېيىشىپ، بىربىرىگە بەس سېلىشقا
باشلاپتۇ. ئاخىر مۈشۈكنىڭ بوينىغا قوڭغۇراق ئېسىپ قويۇشقا جۈئەت قىلىدىغان
بىرەرسىمۇ چىقماپتۇ. ئاڭلاشلارغا قارىغاندا، چاشقانلار تا ھازىرغىچە بۇ ئىشنى
دەتالاش قىلىدىكەن، مۈشۈكلەرمۇ بۇرۇنقىدەك چاشقانلارنى تۇتۇپ يەۋېرىدىكەن. «ياخشى
چارە ئىشنىڭ يېرىمى» دېگەن ماقالمۇ شۇنىڭدىن قالغانىكەن. بېرنادشاۋ ئەنگىلىيىلىك
داڭلىق يازغۇچى. بىر قېتىم، ئۇ موسكىۋاغا زىيارەتكە بارغاندا، بىر قىزچاقنى
ئۇچرىتىپ قالدى. چېچىغا قىزىل لېنتا چىگىۋالغان قىزچاق ئاپئاق ھەم بۇدرۇق بولۇپ،
يوغان كۆزلىرى نۇرلۇق، تولىمۇ ئوماق ئىدى. بۇ قىزچاققا ئامراق بولۇپ قالغان
بېرنادشاۋ ئۇنىڭ بىلەن بىللە خېلى ئۇزاق ئوينىدى. خوشلىشىدىغان چاغدا، بېرنادشاۋ
قىزچاققا دېدى: - ئۆيگە بارغاندا ئاپىڭىزغا ئېيتىشنى ئۇنتۇپ قالماڭ، بۈگۈن سىز
بىلەن بىللە ئوينىغان كىشى دۇنياغا داڭلىق يازغۇچى بېرنادشاۋ، بېرناد شاۋ قىزچاق
ئۆزى بىلەن بىللە ئوينىغان كىشىنىڭ دۇنياغا داڭلىق بىر يازغۇچى ئىكەنلىكىنى
بىلدۈرگىنىدە، چوقۇم خوشاللىقىنى ئىچىگە سىغدۇرالماي قالىدۇ دەپ ئويلىغانىدى.
لېكىن، ئويلىمىغان يەردىن، قىزچاق بېرنادشاۋنىڭ تەلەپپۇزىنى دوراپ دېدى: - سىزمۇ
ئۆيىڭىزگە قايتقاندا ئاپىڭىزغا ئېيتىڭ، بۈگۈن سىز بىلەن بىللە ئوينىغان كىشى سوۋېتت
ئىتتىپاقىلىق قىزچاق ناتاشا. بۇ گەپنى ئاڭلىغان بېرنادشاۋ ئىختىيارسىز ئەندىكىپ
كەتتى. ئۇ شۇ ھامان ئۆزىنى بەك يۇقىرى چاغلىۋەتكەنلىكىنى تونۇپ يەتتى. شۇ ئىشتىن
كېيىن، بېرنادشۋا چوڭقۇر ھاياجان بىلەن دەيدىغان بولدى: ئادەم قانچە زور
مۇۋەپپەقىيەتكە ئېرىشىشىدىن قەتئىينەزەر ئۆزىنى چوڭ چاغلىماسلىقى كېرەك ئىكەن. ھەر
قانداق ئادەمگە باراۋەر مۇئامىلە قىلىشى، مەڭگە كەمتەر بولۇشى لازىم ئىكەن. مانا بۇ
ھېلىقى قىزچاقنىڭ ماڭا بەرگەن تەربىيىسى. ئۇ مېنىڭ ئۇستازىم. بالىلىق دەۋر ئەگرى
توقاي كەتكەن ئېرىققا ئوخشايدۇ. بۇ بۆلەكتىكى تېكىستلەر ئاشۇ ئېرىقتا ئوقچۇپ
ئېقىۋاتقان سۇنىڭ بۇژغۇنلىرىغا ئوخشايدۇ. بۇژغۇنلار بەزىدە خوشال ناخشا ئېيتسا،
بەزىدە كۆڭۈل غەشلىكىنى ئىزھار قىلىدۇ؛ ھەر بىر بۇژغۇندا تەسىرلىك ھېكايىلەر بار.
بۇ بۆلەكتىكى تېكىستلەرنى كۆڭۈل قويۇپ ئوقۇپ، بالىلىق دەۋرنىڭ نەقەدەر گۈزەل ۋە
رەڭدار ئىكەنلىكىنى ھېس قىلايلى. باشلانغۇچ مەكتەپتە ئوقۇۋاتقان چېغىمدا، مومام
تۈگەپ كەتتى. مومام ھايات ۋاقتىدا ماڭا بەك ئامراق ئىدى. مەن قايغۇدىن قۇتۇلۇشنىڭ
ئامالىنى تاپالماي، ھەر كۈنى مەكتەپ مەيدانىنى ئايلىنىپ يۈگۈرىدىغان، ھېرىپ
كەتكەندە چىملىققا ئۆزۈمنى ئېتىپ پۇخادىن چىققۇچە يىغلىۋالىدىغان بولدۇم. قايغۇلۇق
كۈنلەر خېلى ئۇزاق داۋاملاشتى. ئاتا - ئانام ماڭا قانداق تەسەللى بېرىشنى
بىلەلمەيلا قالدى. ئۇلار: «موماڭ ئۇخلاپ قالدى» دەپ مېنى گوللىغاندىن كۆرە، راست
گەپ قىلىشنى لايىق تېپىپ: «موماڭ مەڭگۈ قايتىپ كەلمەسكە كەتتى» دېدى. - قايتىپ
كەلمەسكە كەتتى، دېگەن نېمە گەپ؟ - دەپ سورىدىم مەن دادامدىن. - ۋاقىت ئىچىدىكى
بارلىق شەيئىلەرنىڭ مەڭگۈ قايتىپ كەلمەيدىغان چېغى بولىدۇ، - دېدى دادام
چۈشەندۈرۈپ، - سېنىڭ تۈنۈگۈنۈڭ ئۆتۈپ كەتسە، ئۇ مەڭگە تۈنۈگۈن بولۇپ قالىدۇ، سەن
ھەرگىزمۇ تۈنۈگۈنگە قايتالمايسەن. مەن بۇرۇن ساڭا ئوخشاش كىچىك ئىدىم، ھازىر ئاشۇ
بالىلىقىمغا قايتالمايمەن. بىر كۈنلەرگە بارغاندا، سەنمۇ چوڭ بولىسەن، موماڭغا
ئوخشاش قېرىيسەن. يەنە بىر كۈنلەرگە بارغاندا، سەنمۇ ئۆمرۈڭنىڭ بارلىق ۋاقتىنى
ئۆتكۈزۈپ بولۇپ، موماڭغا ئوخشاش مەڭگە كەلمەسكە كېتىسەن. دادامنىڭ دېگەنلىرى بىر
تېپىشماققا ئوخشايتتى. بۇ تېپىشماق ماڭا «ۋاقىت - ئالتۇندىنمۇ قىممەت، ۋاقىتنى
ئالتۇنغا سېتىۋالغىلى بولمايدۇ» دېگەن سۆزلەرگە قارىغاندىمۇ ۋەھىمىلىك بىلىندى،
«ۋاقىت - ئاتقان ئوق، ۋاقىت ئۇچقاندەك ئۆتەر» دېگەن سۆزلەردىنمۇ بەكرەك
جىددىيلەشتۈردى. شۇنىڭدىن كېيىن، مەن ھەر كۈنى مەكتەپتىن قايتىپ كەلگەندە، ھويلىدا
ئولتۇرۇپ قۇياشنىڭ تاغ كەينىگە ئاستا - ئاستا پېتىپ كېتىۋاتقانلىقىغا قارايدىغان،
«مانا، يەنە بىر كۈن كەلمەسكە كەتتى» دەيدىغان بولدۇم. ئەتە قۇياش قايتا چىقسىمۇ،
بىراق بۈگۈنكى قۇياشنىڭ مەڭگۈ بولمايدىغانلىقىنى چۈشىنىپ يەتتىم. مەن ئۇچۇپ
كېتىۋاتقان قۇشلارنى كۆردۈم، ئۇلار بەك تىز ئۇچىدىكەن. ئەتە ئۇلار مۇشۇ يول بىلەن
قايتا ئۇچۇپ ئۆتەر، لېكىن ئۇ كۈن مەڭگۈ بۈگۈن بولالمايدۇ. ئېھتىمال ئەتە بۇ يول
بىلەن ئۇچۇپ ئۆتىدىغىنى قېرى قۇش ئەمەس، بەلكى بالا قۇش بولىشىمۇ مۇمكىن. تېز
ئۆتۈۋاتقان ۋاقىت مېنى تىتتىت قىلىپلا قالماي، يەنە كۆڭلۈمنى يېرىم قىلاتتى. بىر
كۈنى مەكتەپتىن قايتىپ كېلىۋېتىپ، قۇياشنىڭ ئولتۇراي دەپ قالغىنىنى كۆرۈپ دېدىم:
مەن ئۆيۈمگە قۇياشتىنمۇ تېز قايتالايمەن. ئۇچقاندەك يۈگۈرۈشكە باشلىدىم. مەن
ھاسىرىغىنىمچە ھويلىغا كىرىپ قۇياشقا قارىسام، ئۇنىڭ ئاران يېرىمى ئولتۇرۇپتۇ. مەن
خوشاللىقىمدىن سەكرەپ كەتتىم. شۇ كۈنى مەن قۇياشنى يېڭىۋالدىم. كېيىن، دائىم مۇشۇ
ئويۇننى ئوينايدىغان بولدۇم. مەن بەزىدە قۇياش بىلەن، بەزىدە شامال بىلەن
بەسلىشەتتىم؛ بەزىدە ۋاقىت بىلەن بەسلىشىپ، تەتىللىك تاپشۇرۇقلىرىمنى. ئون كۈندىلا
ئىشلەپ بولاتتىم. 3-يىللىقتا ئوقۇۋاتقان ۋاقتىمدا، ھەمىشە 5-يىللىقتا ئوقۇۋاتقان
ئاكامنىڭ تاپشۇرۇقلىرىنى تالىشىپ ئىشلەپ كېتىدىغان بولدۇم. ھەر قېتىم ۋاقىت بىلەن
بەسلىشىپ ئۇنى يېڭىۋالغىنىمدا، خوشاللىقىمنى قانداق سۈپەتلەشنى بىلەلمەي قالاتتىم.
كېيىنكى 20 يىلدا، مەن بۇنىڭدىنمۇ زور نەپكە ئېرىشتىم. مەن مەڭگۈ ۋاقىتنى
يېڭەلمەيدىغانلىقىمنى بىلسەممۇ، بىراق ئەسلىدىكىدىن بىر قەدەم بولسىمۇ تېز
يۈگۈرگىلى بولىدىغانلىقىنى چۈشىنىپ يەتتىم. بۇ بىرنەچچە قەدەم بەك كىچىك بولسىمۇ،
بىراق ئۇلارنىڭ رولى كىچىك ئەمەسكەن. چۈنكى، ۋاقىت بىلەن ئىزچىل بەسلەشكەندە،
جەزمەن مۇۋەپپەقىيەتكە ئېرىشكىلى بولىدىكەن. ئىيۇل ئېيى يېتىپ كېلەي دېدى. پولشانىڭ
پايتەختى ۋارشاۋادا ئىلگىركى يىللاردىكىگە ئوخشاش، دۆلەت بايرىمىنى قارشى ئېلىش
تەييارلىقلىرى باشلىنىپ كەتتى. بىر مەكتەپنىڭ كۇلۇبىدا بالىلار ئەترىتىنىڭ ئەزالىرى
يىغىن ئېچىپ، دۆلەت بايرىمىلىق پاراتقا قاتنىشىش ئىشىنى مۇزاكىرلىشىۋاتاتتى. بەزى
ئىشلار بىكىتىپ بولۇندى، ئەمما ھەممەيلەننىڭ چىرايىدىن مەسلىھەتلىشىشكە تىگىشلىك
يەنە بىر مۇھىم ئىشىنىڭ بارلىقى بىلىنىپ تۇراتتى. ئۇنىڭغا بىردىنلا ئامراقلىقى
كەلگەن خوجايىن دېدى. - بولىدۇ، ئاپىرىپ سىزىپ باققىن، قاملاشتۇرالمىساڭ ئەدىپىڭنى
بېرىمەن جۇمۇ. ئەتىسى ئەتىگەندە، دېسلو كېچىچە ئىشلەپ پۈتكۈزگەن ۋىۋىسكىنى كۆتۈرۈپ
دۇكانغا باردى. بۇ چاغدا، دۇكان خوجايىنى سالاپەتلىك بىر بوۋاي بىلەن پاراڭلىشىپ
تۇراتتى. خوجايىن ۋىۋىسكىنى كۆرۈپ ھەيرانلىقىدىن تۇرۇپلا قالدى ۋە ئاغزىئاغزىغا
تەگمەي ماختاپ كەتتى: - نېمەدېگەن چىرايلىق! بۇ مېۋە - چېۋىلەر گويا ھازىرلا
ئۈزۈلگەندەك كۆرۈنىدىكەن! - ۋىۋىسكىنى مۇشۇ بالا سىزغانلا بولسا، ھەقىقەتەن چېچەن،
ئەقىللىق بالا ئىكەن، - دېدى ھېلىقى بوۋايمۇ، ئاندىن دېسلوغا قاراپ، - سەن بىز
بىلەن خىزمەت قىلىشنى خالامسەن؟ مەن ساڭا ئىش ھەققى بېرىمەن، سېنى ئوقۇشقىمۇ
ئەۋەتىمەن، سەن ماڭا ئېلان رەسىملىرىنى سىزىپ بېرىسەن. دېسلو ماقۇل بولدى. بۇ كىشى
بىر يىمەكلىك زاۋۇتىنىڭ ماركا لايىھىلىگۈچىسى ئىدى. كېيىن، ئۇ دېسلونى ئۆز
تەربىيىسىگە ئېلىپ، باشلانغۇچ ۋە ئوتتۇرا مەكتەپلەردە ئوقۇتتى، يەنە گۈزەل - سەنئەت
مەكتىپىگە ئوقۇشقا بەردى. دېسلو تىرىشىپ ئۆگىنىپ، ئاخىرى داڭلىق رەسسام بولۇپ
قالدى. ئەپەندىمنىڭ بىر يېزىلىق ئاغىنىسى ئۇنىڭدىن شەھەردىكى دوستىغا خەت يېزىپ
بېرىشنى ئىلتىماس قىلىپتۇ. نەسردىنئەپەندى ئۆزرە ئېيتىپتۇ. - جېنىم بىلەن يېزىپ
بېرەتتىم، بىراق شەھەرگە بېرىشقا ۋاقتىم يوق، - ئەپەندىم، سىزنى شەھەرگە بېرىڭ
دېگىنىم يوق، خەت يېزىپ بېرىڭ دەۋاتىمەن، - دەپتۇ ئاغىنىسى. - رەنجىمەڭ، ئاغىنە،
مەن يازغان خەتنى ئۆزۈمدىن باشقا ئادەم ئوقۇيالمايدۇ . سىزگە خەت يېزىپ بەرسەم،
ئۆزۈم بېرىپ دوستىڭىزغا ئوقۇپ بەرسەم بولىدۇ، - دەپتۇ ئەپەندىم. ئەپەندىمنىڭ
داڭقىنى ئاڭلىغان بىر كىشى كېلىپ دەپتۇ. - ئەپەندىم، دۇنيادا نۇرغۇن ئىشلارغا دۇچ
كەلدىم، قايسىسىنى ئۇنتۇپ، قايسىسىنى يادىمدا ساقلاشنى بىلەلمەي قالدىم، بىر
مەسلىھەت بېرىڭا، - بۇ دۇنيادا باشقا كىشىدىن كۆرگەن ياخشىلىقىڭىز بولسا، ئۆلگۈچە
ئېسىڭىزدە ساقلاڭ؛ باشقىلارغا قىلغان ياخشىلىقىڭىز بولسا، ئۇنى دەررۇ يادىڭىزدىن
چىقىرىپ تاشلاڭ! - دەپ جاۋاپ بېرىپتۇ ئەپەندىم. ئەپەندىم بىر يېقىن دوستىغا خەت
يېزىۋاتقاندا، بىرى ئشىك كەينىگە كېلىپ، ئوغرىلىقچە خەتنى ئوقۇپتۇ. بۇنى سېزىپ
قالغان ئەپەندىم خەتنىڭ ئاخىرىغا يېزىپتۇ: «قەدىرلىك ئاغىنە، ساڭا يازىدىغان
گەپلىرىم كۆپ ئىدى، بىراق نۇمۇسنى بىلمەيدىغان بىر ئادەم كەينىمدە تۇرۇپ خېتىمنى
ئوغرىلىقچە ئوقۇۋاتىدۇ». - ئەپەندىم، ئادەمنى نېمىشقا ھاقارەت قىلىسىز، خېتىڭىزنى
كىم ئوقۇپتۇ؟ - دەپتۇ ھېلىقى ئادەم تېرىكىپ. - ئوغرىلىقچە ئوقۇمىغان بولسىڭىز،
ھاقارەت قىلغانلىقىمنى نەدىن بىلدىڭىز؟ - دەپتۇ ئەپەندىم. بۇرۇنىسىدا، ئىنتايىن ئاچ
كۆز، ھىيلىگەر بىر باي ئۆتكەنىكەن. ئۇ يېرىنى دېھقانلارغا ئىجارىگە بېرىپ، تۈرلۈك
ھىيلە - مىكىرلەر بىلەن ھوسولنىڭ ھەممىسىنى دېگۈدەك ئۆزىنىڭ قىلىۋالىدىكەن. بىر
يىلى، بىر نامرات دېھقان بۇ باي بىلەن ئىجارە توختىمى تۈزگىلى بارماقچى بوپتۇ.
بىراق، ئون ياشلىق ئوغلى: «دادا، بۇ ئىشقا مەن بارايچۇ!» دەپ تۇرۇۋاپتۇ. دېھقان
ماقۇل بوپتۇ. بالا باينىڭ ئۆيىگە بېرىپ، كېلىش مەقسىتىنى ئېيتقاندىن كېيىن سوراپتۇ.
- سىلى بۇ يىل تېرىلىدىغان زىرائەتنىڭ قايسى قىسمىنى ئالىدىلا؟ «دادىسى ھاماقەتنىڭ
مۇشتەك شۇمنى ئەۋەتكىنىنى قاراڭلار، - دەپ كۈلۈپتۇ باي ئىچىدە، - تازا
شىلىۋالىدىغان بولدۇم - دە! - ئاستى قىسمىنى، - دەپتۇ باي توختامغا قول قويۇپ. بالا
قايتىپ كېلىپ، ئائىلىسىدىكىلەرگە بۇغداي تېرىشقا مەسلىھەت بېرىپتۇ. بۇغداي
پىشقاندا، باي ئىجارە يىغىشقا كەپتۇ. دېھقان بۇغدايلىرىنىڭ باشىقىنى ئۈزۈۋېلىپ،
غوللىرىنى بايغا قالدۇرۇپتۇ. باي غەزەپلەنگەن ھالدا ھەيۋە قىلغانىكەن، ئەقىللىق
بالا توختامنى چىقىرىپ كۆرسىتىپتۇ. باي غىڭ قىلالماي قاپتۇ. ئىككىنچى يىلى بالا
يەنە توختام تۈزگىلى بېرىپتۇ. «بۇ قىتىم ئەدىپىڭنى راسا بەرمەيدىغان بولسام» دەپ
ئويلاپتۇ باي ۋە: - بۇ يىل زىرائەتنىڭ ئۈستى قىسمىنىمۇ، ئاستى قىسمىنىمۇ ئالىمەن، -
دەپتۇ. - دېگەنلىرىدەك بولسۇن، - دەپتۇ بالا مېيىقىدا كۈلۈپ. ئۇ قايتىپ كېلىپ،
ئۆيىدىكىلەرگە: «بۇيىل قوناق تېرايلى» دەپتۇ. دېھقان قوناقنىڭ بېشىنى ئۈزۈۋېلىپ،
شېخىنى قالدۇرۇپتۇ. بۇ ھالنى كۆرۈپ ئەرۋايى ئۇچقان باي ۋارقىراپتۇ. - مەن بۇ يىلقى
زىرائەتنىڭ ئۈستى قىسمىنىمۇ، ئاستى قىسمىنىمۇ ئالىمەن دېگەن ئىدىمغۇ! -
دېگەنلىرىدەك بولدى، - دەپتۇ بالا كۈلۈپ تۇرۇپ، - بۇ قوناق شاخلىرىنىڭ ئۈستىمۇ،
ئاستىمۇ شۇ پېتى تۇرمامدۇ! باي بۇ نۆۋەتمۇ ئۇتتۇرۇپ قويغانلىقىنى بىلىپ، دەردىنى
ئىچىگە يۇتۇشقا مەجبۇر بوپتۇ. ئۈچىنچى يىلى بالا يەنە توختام تۈزگىلى بارغاندا،
باي: «ئەمدىغۇ دەردىمنى ئالارمەن» دەپ ئويلاپتۇ ۋە: - يەردىن ئۈنۈپ چىققاننىڭ
ھەممىسىنى ئالىمەن! - دەپتۇ. بالا رازىلىق بىلدۈرۈپ: - دېگەنلىرىدەك بولسۇن، -
دەپتۇ. دېھقان ئوغلىنىڭ مەسلىھىتى بويىچە پىياز تېرىپتۇ. كۈز بولغاندا باي كەپتۇ.
ئۇ ئېتىزلىقتىكى ھەممە پىيازنى يىغىۋالماقچى بوپتىكەن، بالا توختامنى كۆرسىتىپ
تۇرۇپ دەپتۇ. - قارىسىلا، توختامدا سىلى يەردىن ئۈنۈپ چىققاننىڭ ھەممىسىنى ئالىمەن
دېگەن، شۇڭا سىلىگە پىيازنىڭ كۆكى تېگىدۇ، يەر تېگىدىكى پىياز توختام بويىچە بىزگە
قالىدۇ، يەنە ئۇتتۇرۇپ قويغان باي دەردكە چىدىماي ئۆيىگە بېرىپلا ئاغرىپ قاپتۇ ۋە
ئۇزاق ئۆتمەي ئۆلۈپ كېتىپتۇ. يېزىدىكى دېھقانلار ئۇنىڭدىن قۇتۇلغىنىغا خۇش بولۇپ،
ئەقىللىق بالىغا بارىكاللا ئېيتىشىپتۇ. بىز تۇرمۇشتا ھەرخىل مەسىلىلەرگە يولۇقۇپ
تۇرىمىز. مەسىلىلەرگە بولغان قارىشىمىز ۋە ھەل قىلىش ئۇسۇلىمىز ئوخشاش
بولمىغانلىقتىن، كۆپ ھاللاردا نەتىجىسىمۇ ئوخشاش بولمايدۇ. بۇ بۆلەكتىكى
تېكىستلەردە جانلىق، قىزىقارلىق ھېكايىلەر بايان قىلىنغان. ئۇلار بىزگە مەسىلىلەرگە
قانداق قاراش، قانداق ئويلاشنى ئېيتىپ بېرىدۇ. قېنى، بىز كۆپرەك ئوقۇپ، كۆپرەك
ئويلىنىپ باقايلى. تەتىلدە ئاتا - ئانام بىلەن قەشقەرگە بارىدىغانلىقىمنى ئاڭلىغان
مۇئەللىم دېدى. «قايتىپ كەلگەندە بىزگە قەشقەر توغرۇلۇق سۆزلەپ بەرگىن، بولامدۇ»
مەن خوشاللىق بىلەن ماقۇل بولدۇم ۋە سەپەرگە چىقىشتىن بۇرۇن قەشقەرگە دائىر بىر
مۇنچە ماتېرىياللارنى كۆرۈپ، بۇ ھەقتە خېلى ساۋاتلىق بولۇپ قالدىم. قەشقەرۋىلايىتى
شىنجاڭئۇيغۇرئاپتونومرايونىنىڭ غەربىي جەنۇبىغا جايلاشقان، شىمالدا تەڭرىتاغ،
جەنۇپتا قاراقۇرۇتېغى، شەرىقتە تەكلىماكانقۇملۇقى، غەربتە پامىرئېگىزلىكى بىلەن
تۇتىشىدۇ. ئومۇمىي يەر مەيدانى 620مىڭ كۋادراتكىلومېتىر. ۋىلايەت تەۋەسىدە بىر
شەھەر، 11 ناھىيە بار، ئومۇمىي نوپۇسى 3مىليون 500مىڭدىن ئاشىدۇ. ئاھالىسىنىڭ
مۇتلەق كۆپ قىسمىنى ئۇيغۇرلار تەشكىل قىلغان. قەشقەر شەھىرى قەشقەر ۋىلايىتىنىڭ
مەركىزى. ئۇ ئىككىئۈچ مىڭ يىللىق تارىخقا ئىگە قەدىمىي شەھەر. ئۇ نەچچە مىڭ يىللار
داۋامىدا كۆپلىگەن خانلىقلارغا پايتەخت بولغان. تارىختا دېڭىز يولى ئېچىلماستا،
قەشقەر قەدىمكى «يىپەك يولىنىڭ مۇھىم قاتناش تۈگۈنى بولغان. قەشقەرنى ئېلىمىزنىڭ
«غەربىي دەرۋازىسى» دېيىشكە بولىدۇ. ئۇ تاجىكىستان، ئاففانىستان، پاكىستان،
قىرغىزىستان، ھىندىستان قاتارلىق بەش دۆلەت بىلەن چېگرلىنىدۇ. قەشقەر ئېلىمىزنىڭ
پاختا ۋە ئاشلىق ئىشلەپچىقىرىش بازىلىرىنىڭ بىرى. مېۋىلىرىنىڭ تۈرىنىڭ كۆپلۈكى،
تەمىنىڭ ئۆزگىچىلىكى بىلەن يەنە «مېۋە ماكانى» دېگەن نامىمۇ بار. قەشقەردە يەنە
يۈسۈپ خاس ھاجىپ، مەھمۇد قەشقەرى قاتارلىق جاھانغا مەشھۇر ئالىملىرىمىزنىڭ
مەقبەرىلىرى، ھېيتگاھ جامەسى، ئاپئاق خوجا مازىرى، تاششەھەر قاتارلىق يادىكارلىقلار
بار. ئەسلىدە، مامكاپ ئېچىلغان چاغدا ئالتۇن رەڭلىك، يايلاقمۇ ئالتۇن رەڭگە
كىرىدىكەن، مامكاپ گۈلى يۇمۇلۇپ قالغاندا بولسا، ئالتۇن رەڭلىك بەرگلىرىنى
ئورىۋالىدىكەن - دە، يايلاق يېشىل رەڭگە قايتىدىكەن. يايلاق نەقەدەر سۈيۈملۈك،
مامكاپ نەقەدەر قىزىقارلىق - ھە! ئەنە شۇنىڭدىن باشلاپ، مامكاپ بىز ئەڭ ياخشى
كۆرىدىغان گۈلگە ئايلاندى. ئۇ بىز بىلەن بىللە ئۇخلايدىغان، بىللە ئورنىدىن
تۇرىدىغان بولدى. سەھەر كۆپكۆك ئاسمان شۇ قەدەر سۈزۈك. مەھەللنىڭ تەرەپ - تەرەپكە
تۇتاشقان تېرەكلىك يوللىرىدىن توپ - توپ بالىلار مەكتەپكە قاراپ كەلمەكتە. ئۇلارنىڭ
ئوماق چىرايلىرىدا كۈلكە جىلۋە قىلاتتى. قىزىق پاراڭ، شوخ كۈلكىلەر ئەتراپنى
جانلاندۇرىۋەتكەنىدى. بىزنىڭ مەكتەپتە ھەرمىللەت بالىلىرى بىرلىكتە ئوقۇيدۇ.
ئۇلارنىڭ تىلى مىجەز - خاراكتېرى، ئارزۇ - ھەۋەسلىرى ئوخشىمىسىمۇ، ئېجىل - ئىناق
ئۆتۈشىدۇ. مەكتىپىمىز دەل - دەرەخ ۋە رەڭگارەڭ گۈللەرگە تولغان گۈزەل باغقىلا
ئوخشايدۇ. مەكتەپ ھويلىسى رەڭمۇرەڭ، خىلمۇخىل مىللىيچە كىيىملەرنى كىيگەن بەختىيار
بالىلار بىلەن تېخىمۇ گۈزەللىشىپ كېتىدۇ. بىز قەدىردان ئوقۇتقۇچىلىرىمىزغا خۇددى
قۇشلاردەك چۇۋۇرلىشىپ سالام بېرىمىز. مەكتەپنىڭ ھەممىلا نەرسىسى بىزنى سۆيۈندۈرىدۇ.
بىزنىڭ مەكتەپكە قاراڭ: «جىرىڭ - جىرىڭ!» قوڭغۇراق چېلىندى. ئوقۇغۇچىلار سىنىپلارغا
كىردى. مەكتەپ ئىچى تىمتاسلىققا چۆمدى. گويا دەرەخ يوپۇرماقلىرى شىلدىرلاشتىن،
قۇشلار سايراشتىن، كېپىنەك ۋە ھەسەل ھەرىلىرى ئۇچۇشتىن توختاپ، تىمتاسلىققا ھەمدەم
بولۇۋاتقاندەك قىلاتتى. تەنەپپۇستا مەكتەپ ئىچى بىردىنلا جانلىنىپ كەتتى.
ساۋاقداشلارنىڭ بەزىلىرى ناخشا ئېيتىپ ئۇسسۇل ئوينىدى. بەزىلىرى ئارغامچا تاقلىسا،
يەنە بەزىلىرى قوغلاشماق ئوينىدى. قىسقىسى، خىلمۇخىل ئويۇنلار تەنەپپۇس ۋاقتىنى
ئاجايىپ كۆڭۈللۈك قىلىۋەتتى. يەنە تېخى يوپۇرماقلار شىلدىرلاپ، قۇشلار بەس - بەستە
سايراپ، كېپىنەكلەر پىلدىرلاپ ئۇچۇپ، بۇ خوشال كەيپىياتقا جور بولدى.. مانا بۇ
بىزنىڭ مەكتىپىمىز، مانا بۇ بىزنىڭ بىلىم باغچىمىز، كۆڭۈللۈك ئائىلىمىز! بىزنىڭ
تۇرمۇشىمىز خۇددى رەڭگارەڭ رەسىملەرگە ئوخشايدۇ. بىز سىنىپتا دەرس ئوقۇيمىز،
مەيداندا ئويۇن ئوينايمىز؛ پەن - تېخنىكا سارىيىنى ئېكسكۇرسىيە قىلىمىز، ئۆسمۈرلەر
سارىيىدا ئويۇن قويىمىز؛ ئېرىق - ئۆستەڭلەردە سۇغا چۆمۈلسەك، ئورمانلىققا بېرىپ
ئەۋرىشكە يىغىمىز بىز كۆڭۈللۈك تۇرمۇش ئىچىدە كۈنسايىن چوڭ بولماقتىمىز. بىز بۇ
بۆلەكتىكى تېكىستلەرنى ئوقۇش ئارقىلىق، ھاياتنىڭ مول ۋە رەڭدارلىغىنى تېخىمۇ ھېس
قىلالايمىز. بىر پادىشاھ ئۆزى تەختكە چىقىشتىن بۇرۇن يۈز بەرگەن بىر ئۇرۇشنىڭ
ۋاقتىنى ئېنىقلىماقچى بوپتۇ. ئەمما، قايتا - قايتا ھېسابلىغان بولىسىمۇ، يەنىلا
توغرا ۋاقىتنى ئېنىقلىيالماپتۇ. شۇنىڭ بىلەن، ئۇ خەلق بىلەن كېڭەش ئۆتكۈزۈپ دەپتۇ:
«بىز بۇ تارىخنى ئېنىقلاشتا قانداق خاتالاشقان بولساق، كەلگۈسى ئەۋلادلىرىمىزمۇ
شۇنداق خاتالىشىدۇ. شۇڭا، بىز 12 ئاي ۋە ئاسماننىڭ بۇرجىغا ئاساسلىنىپ، ھەر بىر
يىلغا ئات قويايلى، بۇنىڭدىن كېيىن يىل ھېسابى شۇ يىللارنىڭ ئايلىنىشى بىلەن
ھېسابلىنىدىغان بولسۇن». ئۇنىڭ دېگىنىنى خەلىقمۇ قوللاپتۇ. شۇ مۇناسىۋەت بىلەن
پادىشاھ ئوۋغا چىقىپتۇ ۋە مۇھاپىزەتچىلىرىگە ھەممە ھايۋانلارنى ئىلى دەرياسىغا
قاراپ قوغلاشنى بۇيرۇپتۇ. بۇ ناھايىتى چوڭ دەريا ئىكەن. مۇھاپىزەتچىلەر بۇ
ھايۋانلارنى ئىلى دەرياسىغا قاراپ ھەيدەپتۇ. ھايۋانلار جان قايغۇسىدا ئۆزلىرىنى
سۇغا ئېتىپتۇ. ئۇلاردىن 12 خىلى دەريادىن ئۈزۈپ ئۆتەلەپتۇ. ئەنە شۇ 12 ىىل
ھايۋاننىڭ ئىسمى 12 يىلغا ئات قىلىپ قويۇلۇپتۇ. سۇدىن ئەڭ ئاۋۋال چاشقان ئۆتۈپتۇ،
شۇڭا يىل بېشى «چاشقان يىلى» دەپ ئاتىلىپتۇ. ئۇنىڭدىن كېيىن ئۆتكەنلەر تۆۋەندىكى
تەرتىپ بويىچە يىل ئىسمى بولۇپ قاپتۇ: توڭگۇز يىلىغا يەتكەندىن كېيىن، ھېساب يەنە
چاشقان يىلىدىن قايتا باشلىنىدىكەن. كېيىن، كىشىلەر يىللارنىڭ رەت تەرتىپىنى ئنتۇپ
قالماسلىق ئۈچۈن، مۇنداق قوشاق توقۇپتۇ: بۇرۇننىڭ بۇرۇنىسىدا، پاتىمە ئىسىملىك بىر
قىز ئۆتكەنىكەن. ئۇ چىرايلىق بولۇپلا قالماي، يەنە بەك ئەقىللىق ئىكەن. كۈنلەرنىڭ
بىرىدە، پاتىمە بەش دوستى بىلەن ئورمانغا ئوتۇن تەرگىلى بېرىپتۇ. ئۇلار ئوتۇن تېرە
- تېرە كەچ كىرگەننى تۇيماپتۇ. شۇنداق قىلىپ ئۇلار قاراڭغۇدا يولدىن ئېزىپ قاپتۇ.
قىزلارنى ۋەھىمە بېسىپتۇ، بىراق پاتىمە دەپتۇ. - قورقماڭلار، بىز قول تۇتۇشۇپ
ماڭايلى، - ئۇلار مېڭىپتۇ، مېڭىپتۇ. بىر چاغدا يىراقتىن پىلىلدىغان يورۇق
كۆرۈنۈپتۇ، يېقىنلىشىپ قارىسا، بىر يېنىۋاتقان گۈلخان ئىكەن. گۈلخاننىڭ يېنىدا بىر
بەتبەشىرە موماي ئولتۇرغۇدەك. ئۇ قىزلارنى كۆرۈپ ئاغزىنى كالچايتىقىنىچە كۈلۈپ
دەپتۇ. - ھا - ھا، نېمە دېگەن ياخشى! ئالتە سېمىز قىز ئۆز ئايىغى بىلەن ئالدىمغا
كەپسىلەر ئەمەسمۇ! قورسىقىڭلار ئاچقاندۇ؟ بىر ئاز يىمەكلىك بېرەي، موماي ئۇلارغا
ناھايىتى چىرايلىق پىشۇرۇلغان گۆشناننى ئۈلۈشتۈرۈپ بېرىپتۇ. پاتىمە بۇ دوستلىرىغا
پىچىرلاپتۇ: - بۇ جادۇگەر موماي ئىكەن، گۆشناننى يېمەيلى، جادۇگەر موماي سوراپتۇ: -
نېمىشقا گۆشناننى يېمەيسىلەر؟ - بىز ئازراق سۇ ئىچىۋالساق، بۇ گۆشناننى يەپ
بولاتتۇق، - دەپتۇ پاتىمە، - دەريا بويىغا بېرىپ سۇ ئىچىپ كەلسەك بوپتىكەن. -
بولمايدۇ! سىلەر قېچىپ كېتىسىلەر. - ئەمىسە، سۇنى سىز ئەكىلىپ بېرىڭ. بىزنى
ئارغامچا بىلەن باغلاپ قويسىڭىز قېچىپ كېتەلمەيمىز ئەمەسمۇ، - دەپتۇ پاتىمە. شۇنىڭ
بىلەن جادۇگەر موماي ئالتە قىزچاقنى ئالتە ئارغامچا بىلەن باغلاپ، ئالتە
ئارغامچىنىڭ ئۇچىنى تۇتقىنىچە، دەريا بويىغا مېڭىپتۇ. جادۇگەر موماي يىراقلاشقاندىن
كېيىن، پاتىمە ئارغامچىنى ئوت بىلەن كۆيدۈرۈپ ئۈزۈپتۇ - دە، ئۇچىنى دەرەخكە باغلاپ
قويۇپتۇ. ئاندىن، دوستلىرى باغلانغان ئارغامچىلارنىمۇ يېشىپ دەرەخكە باغلاپ قويۇپتۇ
ۋە ئۇلارنى باشلاپ قېچىپتۇ. جادۇگەر موماي قايتىپ كەلسە، قىزلار يوق تۇرغۇدەك. ئۇ:
«خەپ، سەن شۇملارنى!» دەپتۇ - دە، ئەپسۇن ئوقۇپ، قىزلارنىڭ ئالدىغا تىمساھ بار چوڭ
دەريادىن بىرنى پەيدا قىپتۇ ۋە شۇ ياققا قاراپ يۈگۈرۈپتۇ. قىزلار دەريا بويىغا
كېلىپ قاپتۇ، دەريادا ناھايىتى چوڭ بىر تىمساھ ئۈزۈپ يۈرگۈدەك. پاتىمەنىڭ دوستلىرى
قورققىنىدىن يىغلىشىپ كېتىپتۇ. پاتىمە دەپتۇ: - تىمساھ ئاكا، بۇ بەش قىزنى دەريادىن
ئۆتكۈزۈپ قويسىڭىز، مەن سىزگە ئوزۇق بولۇشقا رازىمەن. تىمساھ ماقۇل بوپتۇ. ئۇ بەش
قىزنى مىندۈرۈپ بىر - بىرلەپ دەريادىن ئۆتكۈزۈپ قويۇپ قايتىپ كېلىشىگە، جادۇگەر
موماي يېتىپ كەپتۇ. جادۇگەر قارىسا، قىرغاقتا ھېچكىم يوق، دەرھال تىمساھنىڭ
دۈمبىسىگە مىنىپ دەريادىن ئۆتۈپ قىزلارنى قوغلىماقچى بوپتۇ. سىزگە ئىىشك ئېچىلدى.
دۇنيادا نۇرغۇن يېپىقلىق ئىشىك بار. ھەربىر ئشىكنىڭ ئىچىدە سىزگە ناتونۇش بىر دۇنيا
يوشۇرۇنۇپ ياتىدۇ. بىلىم ئىزدەش، جاھان كېزىش دەل بۇ ئىشىكلەرنى ئېچىش جەريانىدۇر.
بۇ ئىشىكلەرنى ئېچىپ يىڭى شەيئىلەر بىلەن ئۇچرىشىش، بىلىمنىڭ يىڭى دۇنياسىغا قەدەم
قويۇش ئادەمنىڭ يۈرەك تارىنى چەكمەي قالمايدۇ، شۇنداقلا، چەكسىىز شادلىققا
تولدۇردۇ. ئاشۇ ئىشىكلەرنى ئېچىپ كۆرۈش ئىستىكى يوق كىشىلەر روھىي جەھەتتىن نامرات،
ئىرادىسى زەئىپ كىشىلەردۇر. تەبىئىەت بىلەن ھاياتلىق ئوتتۇرسىدىكى ئاجايىپ
مەنزىرىلەر باشقىلارنى ھەيرانلىقتىن مەستخۇش قىلغاندا، بۇنداق كىشىلەر غەپلەت
ئۇيقۇسىدا ئوخلاپ ياتقان بولىدۇ. ئىشىك سىرتىدا زېرىكىشتىن لاغايلاپ يۈرۈشىدۇ.
دۇنيادا ئاچقىلى بولمايدىغان ئىشىك يوق. سىز ۋاقىت ۋە زېھنىڭىزنى ئايىمىسىڭىزلا،
يۈرىكىڭىزدە ‹ ‹ ئىشىك › › ئىچىدىكى دۇنيانى كۆرۈش، چۈشىنىش ئىستىكى لاۋۇلداپ
تۇرسىلا، بۇ ئىشىكلەر ئېچىلماي قالمايدۇ. كۆز ئالدىڭىزدا ئاجايىپ مەنزىرىلەر
نامايان بولىدۇ. مۇشۇ تاقاق ئىشىكلەر ئىچىدە ناھايىتى مۇھىم بىر ئىشىك بار. ئۇنىڭ
ئىسمى ئەدەبىيات. ئەدەبىيات ئىنسانىيەتنىڭ ئەڭ نازۇك ھېسسىياتلىرىنىڭ ئەڭ تەسىرلىك
ۋە مول مەزمۇنلۇق ئىزھارى، ئىنسانىيەت تارىخىنىڭ ئەڭ ئوبرازلىق قايتا ئىپادىلىنىش
شەكلى. بىر مىللەتنىڭ ئەدەبىياتى ماھىيەتتە دەل شۇ مىللەتنىڭ تارىخىدۇر. بىر
دەۋىرنىڭ ئەڭ مۇنەۋۋەر ئەدەبىي ئەسەرلىرى شۇ دەۋىرنىڭ ئىخچام كۆرۈنۈىشى، يۈرەك
ساداسىىدۇر، رەڭگارەڭ ئىجتىمائىي تۇرمۇش كارتىنىسى، شۇنداقلا دەۋر روھىنىڭ ۋە
كىشىلەر ھېسسىياتىنىڭ جەۋھىرى. مۇنەۋۋەر ئەدەبىيئەسەرلەر ئىنسانىيەتنىڭ غايىۋى
تەلپۈنۈشى ۋە ئارزۇئىستەكلىرنى ئەۋلادتىن ئەۋلادقا يەتكۈزىدۇ، ئۇنىڭغا
ئىنسانىيەتنىڭ ھېسسىيات دۇردانىلىرى ۋە پاراسەت جەۋھەرلىرى مۇجەسسەم قىلىنغان،
مۇنەۋۋەر ئەدەبىيئەسەرلەرنى ئوقۇش ئارقىلىق تەبىئەتنى چۈشىنىشىمىز، جەمئىيەتنى
چۈشىنىشىمىز، تارىخنى چۈشىنىشىمىز ۋە تۇرمۇشنى چۈشىنىشىمىز چوڭقۇرلىشىدۇ.
ئەدەبىېئەسەرلەرنىڭ ئادەمگە بولغان تەسىرى يۇشۇرۇن. ئەدەبىېئەسەرلەرنى ئوقۇش بىر
خىل مەدەنىيەت جۇغلاش، ئەقىل جۇغلاش، ھېسىسيات جۇغلاش، بىلىم جۇغلاش جەريانىدۇر.
كۆپلىگەن ئىسىل ئەدەبىي ئەسەرلەرنى ئوقۇش ئادەمنىڭ بىلىمىنى ئاشۇرۇپلا قالماي،
ھېسىسياتنى بېيىتىدۇ. ئەدەبىياتتىن خەۋىرى يوق ھالەتتە مەدەنىيەتلىك، يۇقىرى
ساپالىق ئادەم بولۇشنى قىلىش بىمەنىلىكتۇر. ئەدەبىي ئەسەرلەرنى ئەسلا ئوقۇمايدىغان
ئادەم ‹ ‹ ماگىستىر › ›، ‹ ‹ دوكتور › › ياكى ئۇنىڭدىنمۇ يۇقىرى ئونىۋاندىكى ئادەم
بولسۇن، ئۇ پەقەت ‹ ‹ ئالىي ئەقىلگە ئىگە ياۋايى ئادەم › › دىن ئىبارەت. چۈنكى،
ئەدەبىياتقا يىقىنلىشىش، مۇنەۋۋەر ئەدەبىي ئەسەرلەرنى ئوقۇش مەلۇماتلىق ئادەملەرنىڭ
بىلىمىنى ئاشۇرۇش، تەربىلىنىش سەۋيىسىنى ئۆستۈرۈش، ھېسىسياتىنى موللاشتۇرۇشنىڭ
مۇھىم يۇلى. بۇ، ھەممىگە ئايان بىر ھەقىقەت. مەن بۇرۇن ‹ ‹ ئەدەبىياتقا سۆز › ›
دېگەن ماۋزۇدا بىر پارچە يازغانىدىم. مەن بۇ پارچىدا دىئالوگ ئۇسۇلىدىن پايدىلنىپ
ئەدەبىيات توغرىسىدىكى بەزى قاراشلىرىمنى ئوتتۇرغا قويغانىدىم. تۆۋەندە مەن بۇ
پارچىنى كۆپچىلىككە تەقدىم قىلىمەن. ناۋادا ئۇ ئۆسمۈر كىتابخانلارنىڭ ئەدەبىياتقا
بولغان قىزىقىشىنى قوزغىيالىسا، بۇ مېنىڭ بەختىمدۇر. سەن بىپايان زېمىن، چەكسىز
ئاسمان. سەن — كۆككە تاقاشقان تاغ، دولقۇنلۇق دېڭىزئوكيان. سەن رەڭدار قەلىمىڭ
بىلەن دۇنيادا مەۋجۇت بولۇپ قېلىشى مۇمكىن بولغان بارلىق گۈزەللىكنىڭ سۈرىتىنى
سىزىپ چىقىسەن. ئەڭ ئۇلۇغ نەرسىلەردىن ئەڭ ئاددى نەرسىلەرگىچە، ئەڭ ھەيۋەتلىك
نەرسىلەردىن ئەڭ كىچىك نەرسىلەرگىچە نەزىرىڭدىن قېچىپ قۇتۇلالمايدۇ. قىزغىن
داغدۇغىلارىن ئۇلۇغۋار سۈكۈناتلارغىچە سەندە ئەڭ جانلىق ھالدا ئۆز ئىپادىسىنى
تاپىدۇ. كىشىلەر ئۆيدىن چىقماي تۇرۇپ سەن ئارقىلىق دۇنيانى سەيلە قىلالايدۇ. سەن
كىشىلەرگە كىشىلىك ھاياتنىڭ نەقەدەر رەڭگارەڭ ۋە ئەگرى - توقايلىقىنى ئېيتىپ
بېرىسەن. سەن ئۆز سەھنەڭدە ھەرخىل ئادەملەرنى رول ئالغۇزۇپ، ئۇلارغا يۈرەكنى
لەرزىگە سالدىغان كومېدىيە ۋە تراگېدىيىلەرنى ئوينىتىسەن. نەچچە مىڭ يىل بۇرۇن
ياشىغان كىشىلەرنى تىلغا كىرگۈزۈپ، ئۇلارغا ھېسسىياتلىق ھالدا سۈرەتلىگىلى سالىسەن.
ئەڭ تونۇش زامانداشلىرىڭنى تەسۋىرلەش ئارقىلىق ئۇلارغا ئادەم بېشىدىن ئۆتكۈزۈپ
قېلىشى مۇمكىن بولغان مۇمكىنچىلىكلەرنى سۆزلىتىسەن. ئادەملەر سەندە ئەكىس ئەتكەن
خۇشاللىق ۋە قايغۇ، بەخىت ۋە پاجىئەلەرنىڭ تەسىرىدە بەزىدە شاتلىنىپ كۈلسە، بەزىدە
ئازابلىنىپ ئاچچىق ياش تۆكىدۇ. سەن — ئۇزاق ئۆتمۈش، شۇنداقلا ئەمدىلا ئۆتۈپ
تۈنۈگۈن. سەن — چېكى يوق كەلگۈسى. نېپىز ۋاراقلاردىن پۈتكەن كىتاب ئەگەر بىر
يەلكەنلىك كېمىگە تەمسىل قىلىنسا، سەن ئوقۇرمەنلىرىڭنى شۇ كېمىگە سېلىپ، چەكسىز
ۋاقىت ئېقىنلىرىدا ئەركىن - ئازادە سەيلە قىلدۇرسەن. قىرغاقتىكى مەنزىرىلەر ئادەمنى
چەكسىز زوقلاندۇرىدۇ. سەن رىئاللىقنىڭ ئەكىس ساداسى. سەن ئارزۇ - ئارماننىڭ پاللىدە
چاقنىشى. بىز كۆرۈپ تۇرغان ئاددەتكى تۇرمۇش بىلەن يۈكسەك غايىۋى دۇنيا سېنىڭ
قۇچىقىڭدا گىرەلىشىدۇ. سەن بەزىدە ئۇزاق بىر تارىخنى ئەكىس ئەتتۈرسەڭ، بەزىدە
ئۇنتۇلغۇسىز كىچىك بىر دەقىقىنى تەسۋىرلەپ بېرىسەن. ناۋادا سەن راستچىل، سەمىمىي
بولۇپ، تۇرمۇشنى چىنلىق بىلەن ئەكىس ئەتتۈرسەڭ، ئۆزگىرىشچان تەغدىرنىڭ سۈرىتىنى
سىزىپ بېرەلىسەڭ، كىشىلەر قۇدرىتىڭدىن ھاياجانغا چۆمۈپ، ئىختىيارسىز ئاھ ئۇرۇشىدۇ.
سەن كىشىلەرنىڭ يۈرەك تارىنى چەككۈچى سېھرلىك قول. سەن كىشىلەرنىڭ قەلبىدە
ياڭىرىغان ئەكىس سادا. بۇ دۇنيادىكى ئەڭ تەسىرلىك ئاۋاز. ئىنسان قەلبى گويا بىر
دېڭىز، بۇ دېڭىزنىڭ ئاۋازى شۇقەدەر كۈچلۈك ۋە ئۇلۇغۋاركى، ھېچكىم توسىيالمايدۇ،
ھېچقانداق ئاۋاز ئۇنى دورىيالمايدۇ. سەن بىر سەمىمىي ۋە ۋاپادار دوسىت. سېنى
سۆيگەنلەرگە ئۈن - تىنسىز ئۆزۈڭنى تەقدىم قىلىسەن. ئۇلارنى بىپايان گۈزەل دۇنياغا
باشلاپ كىرىپ، دۇنيادىكى ئەڭ جەزبىدار مەنزىرلەرنى كۆرسىتىسەن، ئۇلارغا كىشىلىك
ھاياتنىڭ ھەقىقىتىنى تونۇتىسەن. كىمكى سەن ىلەن دوست بولۇشنى ئارزۇ قىلسا،
قەلبىڭنىڭ دەرۋازىسىنى ئۇنىڭغا كەڭرى ئېچىپ بېرىسەن. ئۇلار سېنى تاشلاپ كەتمىسىلا،
ئۇلارغا مەڭگۈ ئاسىيلىق قىلمايسەن. تاشلاپ كەتمەيسەن. سەن بىر سىرلىق ئىشىك. سېنى
ئېچىشنى نىيەت قىلغانلارنىڭ ھېچقايسىىسىنى نائۈمىد قالدۇرمايسەن. سېنى
شۆھرەتپەرەسلىكنىڭ پەلەمپىيى قىلىشنى خىيال قىغانلارغا ھەرگىز ئېچىلمايسەن. بىر -
بىرىگە مىنگىشىىپ، ئىرماش - چىرماش بولۇپ كەتكەن تاغلار ئارسىدا كېتىۋاتىمىز.
قەدىمكى كارۋانلارنىڭ ناخشىلىرىدا كۈيلەنگەن ئەنە شۇ ئېگىز چوققىلار ئۆز باغرىنى
يېرىپ موكىدەك ئۆتۈشۈپ يۈرگەن زامانىۋى ماشىنا - ئاپتوبۇسلارغا سۈكۈت ئىچىدە
تىكىلىپ تۇرۇشىدۇ. باش ئۈستىڭىزدىلا ئېسىلىپ تۇرغان بۇ تاغلار گويا جاھاننىڭ بارچە
سۈر - ھەيۋىسىنى، ئاجايىپ - غارايىپ رەڭ ۋە كۆرۈنۈشلىرىنى ئۆزىگە
مۇجەسسەملەشتۈرۈۋالغاندەك. زەن قويۇپ قارايدىغان بولسىڭىز، ھەربىر چوققىدا بىر
تونۇش ئوبرازنى كۆرىسىز. بىرلىرى بېشنى ساڭگىلىتىپ، كۆزلىرىنى يۇمۇپ، ھەسىرەت
ئىچىدە ئۆرە تۇرغان قېرى بوۋايغا ئوخشىسا، گاھى بىرلىرى بېشىغا ئوتۇغان تاقىغان،
ئۇچىسىغا ئالتۇن ھەللىك ئۇزۇن كۆڭلەك كىيگەن خانىشقا، يەنە بىرلىرى يوغان گۈرزە
تۇتۇۋالغان زەبەردەس پالۋانلارغا ئوخشايدۇ. يەنىمۇ يىراققا نەزەر سالدىغان
بولسىڭىز، كۆككە تەلپۈنۈۋاتقان بۈركۈت، بالىسىنى باغرىغا بېسىپ ياتقان ئانا،
قاتارلىشىپ بىر - بىرگە ئارتىلىپ ياتقان لەكمىڭ - لەكمىڭ ئەجدىھا، شىر، يولۋاسلار
كۆز ئالدىڭىزغا كېلىدۇ.… گويا ئۇلارنىڭ ھەممىسى قانداقتۇر بىر تىلسىماتنىڭ
كارامىتىدىن بىر دەقىقە ئىچىدە تۇرغان يېرىدە تاشقا ئايلىنىپ كەتكەندەك، ئاخىر بىر
كۈنى تىلسىمات يېشىلىپ، ئەسلىگە كەلگەندە ئورۇنلىرىن تۇرۇپ يوللىرىغا راۋان
بولدىغاندەك، ‹ ‹ ئۇھ، ئەجەپ قاتتىق ئۇخلاپتىمەنا! › › كاپىتان خارۋېر ھەر
قېتىمقىدەك جەنۇبىي ئامپتوندىن غەربىي گېئىن ئارىلىغا قاراپ يولغا چىققان ئىدى.
قاراڭغۇلۇق قوينىغا چۆككەن دېڭىزنى نېپىز بىر قەۋەت تۇمان قاپلىغانئىدى. كاپىتان ‹
‹ نورماندى › › ناملىق بۇ پاراخوتنى ئېھتىيات بىلەن ھەيدەپ كېتىۋاتاتتى، يولۇچىلار
شېرىن ئۇيقۇدا ئىدى. ‹ ‹ نورماندى › › ناملىق بۇ پاراخوت ئېنگىلز دېڭىزى بوغۇزىدىكى
ئەڭ چوڭ ھەم ئەڭ چىرايلىق پوچتا پاراخوتلىرىنىڭ بىرى بولۇپ، ئالتە يۈز توننايۈك
كۆتۈرەلەيتتى. ئۇنىڭ ئۇزۇنلۇقى ئىككى يۈز يىگىرمە چى، كەڭلىكى يىگىرمە بەش چى
كېلەتتى. دېڭىزچىلار ئۇنى ئەڭ ‹ ‹ ياش › › پاراخوت دەپ ئاتشاتتى. چۈنكى ئۇ 1863 –
يىلى ياسالغان بولۇپ، ئەمدىلا يەتتە ياشقا كىرگەنىدى. تۇمان بارغانسېرى
قۇيۇقلىشىشقا باشلىدى. پاراخوت جەنۇبىي ئامپىتون دەرياسىدىن چىقىپ پايانسىز دېڭىزغا
بۆسۈپ كىردى. ئىكويى تاغ تىزمسىغا ئالاھازەل ئون بەش دېڭىز مىلى قالغانىدى. پاراخوت
بىر خىلدا ئىلگىرىلەۋاتاتتى. تاڭ سەھەر سائەت تۆتلەر ئىدى. شۇنداقتىمۇ تۈن
قاراڭغۇلۇقى تېخى تارقىمىغان بولۇپ، پاراخوت ماچتىسىنىڭ ئۇچىنى ئاران ئىلغا قىلغىلى
بولاتتى. ‹ ‹ نورماندى › › غا ئوخشاش بۇ خىلدىكى ئەنگىلىيە پاراخوتلىرىنىڭ كېچىدە
يۈرۈشىدىن ئەنسىرەشنىڭ ھاجىتى يوق ئىدى. توساتتىن، قۇيۇق تۇمان ئارىسىدا تاغ
چوققىسىغا ئوخشىشىپ كېتىدىغان بىر قارا كۆلەڭگە پەيدا بولدى. كاپىتان سىنچىلاپ
قارىۋىدى، قاپقارا پاراخوت بېشىنىڭ كېچە قاراڭغۇلۇقىنى بۆسۈپ، ئۆزى تەرەپكە دولقۇن
يېرىپ كېلىۋاتقانلىقىنى كۆردى. ئۇ چاقپەلەك بىلەن ھەرىكەتلىنىدىغان ‹ ‹ مارى › ›
ناملىق چوڭ پاراخوت بولۇپ، ئودېسسادىن بەش يۈز توننا بۇغداي بېسىپ يولغا چىققانىدى.
ئۇ ئۈستىگە بېسىلغان ئېغىر يۈكنى ئېلىپ شىددەت بىلەن ئۇدۇل ‹ ‹ نورماندى › › ناملىق
پاراخوتقا قاراپ كېلىۋاتاتتى. ئىككى پاراخوتنىڭ بىر – بىرىگە قاتتىق سوقۇلۇشى قاش
بىلەن كىرپىكنىڭ ئارىسىدىكى ئىش بولۇپ قالغانىدى. بۇ بالايىئاپەتتىن قېچىپ قۇتۇلۇش
مۇمكىن ئەمەس ئىدى. شۇ ئەسنادا، ئۇنىڭغا قۇيۇق تۇمان ئارىسىدىن نۇرغۇن پاراخوت
يۇپۇرۇلۇپ كېلىۋاتقاندەك تۇيۇلۇپ كەتتى. كېشىلەر نېمە ئىش بولغانلىقىنى ئاڭقىرىپ
بولغۇچە دېڭىزغا چۈشۈپ بېلىقلارغا يەم بولۇپ كېتەتتى. شىددەت بىلەن ئالغا
ئىلگىرىلەۋاتقان ‹ ‹ مارى › › ناملىق پاراخوت ‹ ‹ نورماندى › › ناملىق پاراخوتنىڭ
بېقىنىغا كېلىپ سوقۇلۇپ، ئۇنىڭدىن چوڭ بىر تۆشۈك ئېچىۋەتتى. قاتتىق سوقۇلۇشتىن
زەخمەت يېگەن ‹ ‹ مارى › › ناملىق پاراخوتمۇ توختاپ قالدى. ‹ ‹ نورماندى › › ناملىق
پاراخوتتا يىگىرمە بەش نەپەر دېڭىزچى، بىر كۈتكۈچى ئايال، ئوتتۇز بىر نەپەر يولۇچى
بار ئىدى. ئۇلارنىڭ ئارىسىدا ئون ئىككى ئايال بار ئىدى. پاراخوت دەھشەتلىك
زىلزىلىدىن قاتتىق چايقىلىشقا باشلىدى. كۆزنى يۇمۇپ ئاچقۇچە يېرىم يالىڭاچ ئەرلەر،
ئاياللار، بالىلار ھەممىسى پالوبىغا يۈگۈرۈپ چىقىشتى. ئۇلار چىرقىرىغىنىچە ئۇياقتىن
– بۇياققا يۈگۈرەيتتى، داد – پەرياد ئۇراتتى، ھەممە كىشى قاتتىق ساراسىمىگە
چۈشكەنىدى. دېڭىز سۈيى پاراخوتنىڭ ئىچىگە شارقىراپ قۇيۇلۇۋاتاتتى. دېڭىز سۈيىنىڭ
ئۇرۇلۇشىدىن پاراخوت ئوچىقىدىن چارسىلدىغان، پىژىلدىغان ئاۋازلار چىقاتتى.
پاراخوتتا پالوبا بىلەن تېشىلگەن يەرنى ئايرىۋەتكۈدەك بىرەر پارچە نەرسىمۇ يوق
ئىدى، قۇتقۇزۇش چەمبىرىكىمۇ يېتىشمەيتتى. كاپىتان خارۋېر قوماندانلىق سۇپىسىدا
تۇرۇپ ئۈنلۈك ۋارقىردى: - ھەممىڭلار تىنىچلىنىڭلار! گېپىمگە قۇلاق سېلىڭلار!
قۇتقۇزۇش قولۋاقلىرىنى پەسكە چۈشۈرۈڭلار. ئاياللار، بالىلار ئەڭ ئالدىدا چىقسۇن.
دېڭىزچىلار ئەڭ ئاخىرىدا چىقىڭلار. ئاتمىش ئادەمنى چوقۇم قۇتۇلدۇرۇش لازىم!
ئەمەلىيەتتە بۇ پاراخوتتا ئاتمىش بىر ئادەم بار ئىدى. شۇ دەقىقىدە ئۇ ئۆزىنى
ئۇنتۇغانىدى. دېڭىزچىلار دەرھال قۇتقۇزۇش قولۋاقلىرىنى يېشىپ پەسكە چۈشۈردى، ھەممە
ئادەم ئۇنىڭغا ئۆزىنى ئاتتى، بىر – بىرىنى ئىتتىرىشىپ، قىستىشىپ، قولۋاقلارنى
ئۆرۈۋەتكىلى قىل قالدى. باش ياردەمچى ئوئاكېلېي بىلەن ئۈچ سەركار چۆچۈپ كەتكەن
كىشىەرنى تىنىچلاندۇرۇپ، ئۇلارنى تەرتىپ بىلەن قۇتقۇزۇش كېمىسىگە چىقارماقچى بولۇپ
كۈچەيتتى. ئەمما تۇيۇقسىز كەلگەن بالا – قازادىن ئېلىشىپ قالايلا دېگەن كىشىلەرنىڭ
ئۇلارنىڭ گېپىگە قۇلاق سالغۇدەك راغدى يوق ئىدى. دېمىسىمۇ، تېخى نەچچە مىنۇتنىڭ
ئالدىدا كىشىلەر شېرىن ئۇيقۇ قوينىدا ئىدى، مانا ئەمدى ھايات – ماماتى قىلنىڭ
ئۈستىدە تۇرسا، ئۇلار ئەقلى - ھوشىنى يوقاتماي مۇمكىنمۇ! دەل شۇ پەيتتە، كاپىتاننىڭ
كۈچلۈك ئاۋازى داد – پەرياد ئاۋازلىرىنى بېسىپ چۈشتى. كچېە قاراڭغۇلۇقىدا قىسقا،
ئەمما يۈرەككە تېگىدىغان سۆزلىشىش ئاڭلاندى. - مېخاينىك لوككېر قەيەردە؟ - كاپىتان،
مېنى چاقىردىڭىزمۇ؟ - ئوچاق قانداق بولدى؟ - سۇ كىرىپ كەتتى. - توختاپ قالدى.
كاپىتان يەنە چاقىردى: - باش ياردەمچى ئوئاكېلېي! - مانا مەن! - دەپ جاۋاب بەردى
باش ياردەمچى. - پاراخوت چۆكۈپ كېتىشكە يەنە قانچە مىنۇت بار؟ - ۋاقىت يەتكۈدەك، -
دېدى كاپىتان، - بىرمۇ ئادەم قالماي ھەممىسى قۇتقۇزۇش قولۋىقىغا چىقسۇن. باش
ياردەمچى ئوئاكېلېي، تاپانچىڭىز بارمۇ؟ - قايسى ئەر نۇمۇس قىلماي ئاياللارنىڭ
ئالدىغا ئۆتۈۋالسا، ئېتىۋىتىڭ! ھەممە ئادەم جىمىپ قالدى. ئۇنىڭ ئىرادىسىگە خىلاپلىق
قىلدىغان بىرمۇ ئادەم چىقمىدى. بىر ئۇلۇغ روھ ئۆزلىرى ئۈستىدىن ھۆكۈمرانلىق
قىلىۋاتقاندەك ھېسسىياتقا كېلىپ قالغانىدى. ‹ ‹ مارى › › ناملىق پاراخوتتىكىلەرمۇ
قۇتقۇزۇش قولۋاقلىرىنى چۈشۈرۈپ، بالا – قازاغا يۇلۇققان كىشىلەرنى توشۇۋاتاتتى.
قۇتقۇزۇش خىزمىتى رەتلىك كېتىۋاتاتتى. ھېچقانداق تالاش – تارتىش ياكى جېدەل – ماجرا
يۈز بەرمىدى. دۇنيانىڭ ئىشلىرى شۇنداق، قەيەردە رەزىل شەخسيەتچىلەر بار بولسا،
ئۆزىنى قۇربان قىلىپ باشقىلارنى قۇتقۇزدىغان مەرد – مەردانىلەرمۇ كۆكرەك كېرىپ
ئوتتۇرىغا چىقىدۇ. خارۋېر يەنىلا ئۆزىنىڭ كاپىتانلىق ئورنىدا مەزمۇت تۇرۇپ، ھەممىگە
قوماندانلىق قىلىۋاتاتتى. كىچىككىنە بىر ئىش ياكى بىرەر ئادەممۇ ئۇنىڭ نەزەرىدىن
قېچىپ قۇتۇلالمايتتى، ئۇ زەررىچە بىخەستەلىك قىلمايتتى. ئەس – ھوشىنى يوقۇتۇپ نېمە
قىلارىنى بىلمەيۋاتقان ئادەملەر ئالدىدا ئۇ ئۆزىنى شۇنچىلىك تەمكىن تۇتۇۋاتاتتىكى،
خۇددى بالا – قازاغا ئۇچرىغان ئادەملەرگىلا ئەمەس، بەلكى بالا – قازاغىمۇ بۇيرۇق
چۈشۈرۋاتقاندەك، ھەتتا ھالاكەتكە ئۇچرىغان پاراخوتمۇ ئۇنىڭ ئورۇنلاشتۇرۇشىغا
بويسۇنۇۋاتقاندەك قىلاتتى. بىردەمدىن كېيىن ئۇ يەنە ۋارقىرىدى: - كلېيمىننى
قۇتقۇزۇپ چىقىڭلار! كلېيمىن پراكتىكانت ماتروس بولۇپ، تېخى بىر بالا ئىدى . پاراخوت
چوڭقۇر دېڭىز تەكتىگە ئاستا – ئاستا چۆكمەكتە. كىشىلەر سۈرئىتىنى تېخىمۇ تېزلىتىپ،
‹ ‹ نورماندى › › ناملىق پاراخوتتىن ‹ ‹ مارى › › ناملىق پاراخوتقا چىقماقتا. - تېز
بۇلۇڭلار! كاپىتان يەنە خىتاب قىلدى. يىگىرمە مىنۇت توشتى، پاراخوت پۈتۈنلەي چۆكۈپ
كەتتى. باشتا پاراخوتنىڭ بېشى، ھەش - پەش دېگۈچە گەۋدىسى قوشۇلۇپ پاراخوت قۇيرۇقى
چۆكتى. كاپىتان خارۋېر يەنىلا ئۆز ئورنىدا مەزمۇت تۇراتتى. ئۇ بىرەر قول
ئىشارىسىمۇ، بىرەر ئېغىز گەپمۇ قىلمىدى، تىك تۇرغىنىچە پاراخوت بىلەن بىللە دېڭىزغا
غەرق بولدى. كىشىلەر شۇملۇق پۇراپ تۇرىدىغان نېپىز بىر قەۋەت تۇمان ئىچىدە بۇ قارا
رەڭلىك ھەيكەلنىڭ دېڭىز قوينىغا كىرىپ كەتكىنىگە قاراپ تۇردى. كاپىتان خارۋېرنىڭ
ھاياتى مانا مۇشۇنداق ئاياغلاشتى. ئېنگىلىز دېڭىزى بوغۇزىدىكى ھېچقانداق دېڭىزچى
ئۇنىڭدەك ئۇلۇغلانمىغانىدى. ئۇ ئۆمۈر بويى ئۆز خىزمىتىگە سادىق بولۇپ، ئادىمىيلىك
بۇرچىنى ئادا قىلىپ كەلگەنىدى. ئۆلۈم ئالدىدا ئۇ قەھرىمانلاردىن بولۇش ھوقۇقىنى
ئىشلەتتى. شېئىر ھەم ھۆسنى، ھەم پۇرىقى ھەم تىكىنى، ھەم يوپۇرمىقى بار گۈلگەئوخشاپ
كېتىدۇكى، ھۆسنى بىلەن دىلنى ئۆزىگە مەپتۇن قىلار؛ پۇرىقى بىلەن جانغا راھەت ۋە
ھۇزۇر بەخىش ئېتەر؛ ئىشىق ئەھلى ئۇنى قۇلاققا قىسسار ياكى گۈلدەستە ياساپ جانانىغا
تۇتار؛ ياۋا توڭگۇز تومشۇقىنى تەگكۈزسە، تىكىنى ئۇنڭ بۇرنىغا سانجىلار، ھەتتا
بۇرنىنى قانىتار. بىر كۈنى كەچتە دېڭىز بويىغا ئاتلىق ساياھەتكە كەلگەن بىر ئادەم
يول بويىدىكى بىرمېھمانخانىغا چۈشتى. ئۇ كىشىلەنىڭ سەمىمىيتىگە ۋە جەمئىيەتنىڭ
ئامانلىقىغا ئىشىنىپ، ئېتىنى دەرۋازا ئالدىدىكى دەرەخكە باغلاپ قويۇپ مېھمانخانىغا
كىرىپ يېتىپ قالدى. تۈن نىسبىدە بىر ئوغرى ئۇنىڭ ئېتىنى ئوغرىلاپ كەتتى. ئەتىسى
ئەتىگەندە ئېتىنىڭ يوقالغانلىقىنى بىلگەن ساياھەتچى ھەسرەت چېكىپ قاقشاپ كەتتى.
مۇشۇنداق يەردىمۇ ئوغرى بارلىقىدىن قاتتىق ئەپسۇسلاندى. شۇ چاغدا بىر توپ
ساياھەتچىلەر ئۇنى ئورىۋېلىپ پاراڭ باشلىدى: _ _ نېمانچە ئەخمەق ئادەم بۇ، ئاتنى
ئاتخانىنىڭ سىرتىغىمۇ باغلاپ قويغان بارمۇ؟ - دېدى بىرى. _ _ تېخىمۇ دۆتلۈك قىلىپ
ئېتىڭىزنى چۈشەپمۇ قويماپسىز، - دېدى يەنە بىر. _ _ دېڭىز بويىغا ئاتلىق ساياھەتكە
كېلىشتىن ئارتۇق ھاماقەتلىك يوق، - دېدى ئۈچىنچىسى. _ _ ئاتلىق يۈرىدىغان ئادەم
ھۇرۇن، يول مېڭىشقىمۇ ئېرىنىدىغان لەقۋا ئادەمدۇر، دېدى تۆتىنچىسى. ساياھەتچى چۆچۈپ
كەتتى ۋە ۋارقىراپ كەتتى: _ _ خالايىق، ئېتىمنىڭ يوقالغانلىقى تۈپەيلىدىن ھەممىڭلار
تۇشمۇ - تۇشتىن مېنى ئەيىبلەۋاتىسىلەر. ئەجىبا، نومۇسىسز ئوغرىنى ئەيىبلەيدىغانلار
يوقمۇ؟! كۈنلەرنىڭ بىردە ئېتىزلىقتا ئىشلەۋاتقان دېھقان ناھايىتى ئېسىل بىر مەرمەر
ھەيكەلنى قېزىۋالدى – دە، ئۇنى ئاپىرىپ خىلمۇخىل ئەتىۋارلىق سەنئەت بۇيۇملىرىنى
يىغىپ ساقلىغۇچىغا ساتتى. يىغىپ ساقلىغۇچى بۇ ئېسىل ھەيكەلنى يۇقىرى باھادا
سېتىۋالدى. دېھقان قايتىپ كېتىۋېتىپ پۇلغا قاراپ ئويلانغىنىچە ئۆز - ئۆزىگە
پىچىرلىدى: ‹ ‹ بۇ پۇل ئىنسان ھاياتىنى قالتىس ھۇزۇر – ھالاۋەتكە سازاۋەر قىلىدۇ!
مەرمەر ھەيكەل ئۆلۈك نەرسە، ئۇ يەر ئاستىدا مىڭ يىللاپ ياتقان، ئۇنى ھېچكىم
خىيالىغا كەلتۈرۈپ باقمىغان. نېمە ئۈچۈن بىراۋ ئۇنى شۇنچە جىق پۇلغا سېتىۋالىدۇ؟ ›
› ئەمما يىغىپ ساقلىغۇچى ھەيكەلگە قاراپ زوقلىناتتى. ئۇمۇ ئۆز - ئۆزىگە
پىچىرلايتتى: دەريا ئۇپقۇنلىرى قىرغاقلاردىن ھالقىپ يىراق جايلارغا ئېقىپ
بارالايدۇ. ئۇلارمۇ ئاخىر ئۇپقۇنلارغا قوشۇلىدۇ، ھەتتا يىراق دېڭىزلارغىچە ئېقىپ
بارىدۇ. كىشىلەرنىڭ مۇھەببىتىمۇ شۇنداق. بۈيۈك مۇھەببەت پۈتكۈل دۇنياغا مەنسۇپ، ئۇ
ھەممىلا ئادەمگە بەخت ئاتا قىلىدۇ. يەنە ئاددىي كىشىلەرنىڭمۇ مۇھەببىتى بار. ئۇ
سىرغىپ ئېقىۋاتقان سۇغا ئوخشايدۇ، ئۇمۇ كىشىلەرگە گۈزەللىك بەخش ئېتىدۇ. ئۇنىڭدىن
باشقا شۇنداق بىرخىل مۇھەببەتمۇ باركى، ئۇ خۇددى قۇرتلاپ كەتكەن توختام سۇغا ئوخشاش
كىشىلەر كۆڭلىنى غەش قىلىدۇ. قار ئاق توزغاندەك يۇمشاق ئىدى. ھەتتا ياۋا توشقانمۇ
ئۇنى كۆكرىكى بىلەن تۈرتۈپ، ئۇنىڭ ئاستىدا بىمالال ھەرىكەتلىنەلەيتتى. يىراق
جايلاردىن ئۇچۇپ كەلگەن قۇشلار قارايغان چىلان تۇپراقلاردىن ئۇزۇق ئىزلەشكە
باشلىدى. يامغۇر شەربىتىگە تويۇنغان قېيىنلار خۇشاللىقتىن تارام - تارام ياش تۆكتى.
تامچىلىرى قار ئۈستىگە تېمىۋېتىپ، ئۇنى ھەرە كۆنىكىگە ئوخشىتىپ قويدى. يول
ئۈستىدىكى مۇزلار پارە - پارە بۇلۇپ كەتكەن ئەينەك پارچىلىرىدەك چېچىلىپ ياتاتتى.
شۈرمەللەر ئېقىتۋاتقان دەريا قىرغاقلىرى بوشىشىپ، ئولتۇرۇشقا باشلىدى... كېچىدە بىر
ياۋا توشقان بۇ يەردىن ئۆتتى - دە، يىراقلارغا يۈگۈرۈپ كەتتى. ئۇنىڭ ئاياق ئىزلىرى
يۇمشاققىنە ساھىل ئۈستىگە گۈل بولۇپ چېكىلەتتى. گۆدەك چاغلىرىمدا چوڭىيىپ كېتىشتىن
بەكمۇ ئەنسىرەيتتىم. خېلى ياشلارغا كىرىپ قالغان بولساممۇ ھېچقانچە ئىشنى ۋۇجۇدقا
چىقىرالمىدىم، دەپ ئويلايتتىم. شۇڭا، ئون يەتتە ياشقا كىرگۈچە ئۆز - ئۆزەمنى
ئەيىپلەپ يۈردۈم. لېكىن، ئون يەتتە ياشتى ھالقىغاندا كىشىلەر مېنى &lt; &lt; پاھ،
قالتىس جۇمۇ سەن! &gt; &gt; دەپ ماختىشدىغان بولدى. مەنمۇ سۇدەك ئۆتۈۋاتقان
ئۆمرۈمگە پۇشايمان قىلمايدىغان، ھەتتا قانچە چوڭايسام كىشىلەر مېنى سۇنچە
ماختايدىغان ئوخشايدۇ، دەپ ئويلاپ قالدىم. بىر كۈنى فېنيە چوڭ ئانام ماشىنىمىزنى
يۇيۇشقا ياردەملەشتى. ئۇ سۇ بىلەن يۇيدى، مەن بۇغا تېرىسى بىلەن سۈرتتۈم. مەن خېلى
كۈچىنىپ ئىشلىدىم. ئۇ پىشانەمدىكى تەرنى كۆرگەن بولسا كېرەك، مەندىن: _ _ قانچە
ياشقا كىردىڭىز؟ _ _ دەپ سورىدى. مەن ئۇنىڭغا بىر پەس تىكىلىپ قاراپ قالدىم ھەمدە
تەشۋىشلىنىشكە باشلىدىم. چۈنكى، مەن ئاددىي ئادەمنىڭ كۆزلىرىدىن تەبىئەتنىڭ
رەھىمسىز قانۇنىيىتىنى كۆرگەنىدىم. مەن شۇندىلا قىلبىمدىكى لىرىك كۈيلەر ۋە
شائىرانە تۇيغۇلارنىڭ مېنى قۇتقۇزۇپ قالالمايدىغانلىقىنى ھېس قىلىپ يەتتىم. مەن
ئۇنىڭ كۆزلىرىگە تىكىپ تۇرۇپ قالدىم. كۆڭلۈم پاراكەندە ئىدى. مەن ئوڭايسىزلانغان
ھالدا ئۇنىڭ سوئالىنى تەكرارلىدىم: _ _ نەچچە ياشقا كىردىڭىز دەمسىز؟ _ _ شۇنداق
بېگىم، _ _ دېدى چوڭ ئانام، _ _ نەچچىگە كىردىڭىز دەۋاتىمەن. مەن كۆڭلۈمدىكى
تەھلىكىنى ئاخىر بېسىۋالدىم؛ _ _ قانچە ياشقا كىردىڭىز دەمسىز؟ ئۆزىڭىز دەپ بېقىڭ،
مانا مەن ئاتتەك ئىشلەۋاتىمەنغۇ. _ _ مەنغۇ كۆرۈپ تۇرۇۋاتىمەن، دېدى ئۇ، _ _ يامان
ئەمەس ئىشلەۋاتىسىز. لېكىن، زادى قانچە ياشقا كىرگەنسىز؟ _ _ ئاتتەك
ئىشلەۋاتىمەنغۇ، _ _ دېدىم مەن سۆزۈمنى تەكرارلاپ، _ _ كىشىلەر ئاتقا مېنىۋالغاندا،
ئاتنىڭ چىشىغا قارىمايدۇ - دە. _ _ جايىدا گەپ قىلدىڭىز! _ _ فىنيە قوشۇلغانلىقىنى
بىلدۈردى. بىزنى سۆھبىتىمىزمۇ شۇنىڭ بىلەن تامام بولدى. ئۇ ئويلانغان ھالدا ئۆز -
ئۆزىگە: &lt; &lt; ئادەم دېگەن ئۇزاقراق ياشىسام دەيدۇ - دە &gt; &gt; دەپ قويدى.
كەچكى شەپەق چوغدەك جۇلالىنشقا باشلىدى. دەرەخ ئۇچىلىرىغا قىزغۇچ جىيەك تۇتۇلۇش
بىلەن تەڭ دەرەخ تۈۋى سۇس قاراڭغۇلۇققا مەھكۇم بولدى. بوزىراڭغۇ تولۇنئاي قۇياش
ئورنىدا بالقىشقا تەييارلانىدى. دەرەخنىڭ ئەڭ ئېگىز شاخلىرىغا چۈشۈپ تۇرغان
قۇياشنىڭ ئاداققى نۇرىمۇ خىرەلەشتى. رەسسام قەلىمىنى تاشلاپ قويدى. _ _ سىزىپ بولاي
دېگەنىدىم، _ _ دېدى ئۇ. _ _ ئەمدى نېمە قىلىسىز؟ _ _ دەپ سورىدۇق بىز. _ _ ھېچ ئىش
يوق، يەنە بىر كەچكى قۇياشنى كۈتمەي ئامال يوق. ئازراقلا ۋاقىت بولسا يېتەتتى. _ _
لېكىن، بۇ پەيت تەبىئەت دۇنياسى بىلەن ئەبەدىي خوشلاشقان تۇرسا، ئۇ يەنە قايتىپ
كېلەمدۇ. _ _ ئەلۋەتتە كەلمەيدۇ. لېكىن، شۇنىڭغا ئوخشىشپ كېتىدىغان پەيتلەر يەنە
كېلىدۇ. مەن ئۆتكەن ئىشلارنى ئەسلىيەلەيمەن ھەم ئېسىمدە مەڭگۈ ساقلىيالايمەن. _ _
بۇ مۇمكىنمۇ؟ _ _ ئەلۋەتتە! ئەگەر بىر ئادەم ئۆتۈپ كاتكەن شۇنداق پەيتلەرنى ئېسىدە
ساقلىيالمىسا ئۇنى ئادەم دېگىلى بولمايدۇ؟ بىر يىلتىزدىن ئىككى تۈپ قېيىن نوتىسى
باغجانلاپ چىقتى. ئۇلار ئادەم بويى ئۆرلىگەندە ئىنسان تىلىدا پارڭلاشتى: _ _ مەن
ئۆزۈن ئارزۇيۇم بويىچە ياشايمەن، _ _ دېدى بىرى. _ _ مەنمۇ شۇنداق! _ _ دېدى يەنە
بىرىمۇ. ئۇلار شۇ سەبىي ئارزۇلىرى بويىچە ئايرىلىپ كېتىشتى. بىر يىلتىزدىن ئوزۇق
ئېلىپ ئېگىز ھەم بۈك-باراقسان دەرەخ بولۇپ يېتىشتى. ئۇلار يۈز يازنى بېشىدىن
ئۆتكۈزدى. بورانلار قۇم ۋە توزانلارنى ئۇچۇرۇپ كېلىپ ئۇلارنىڭ ئارسىنى تىندۇردى.
ئۇلارنىڭ ئارسىدىكى بۇ ئۈنۈمسىز تۇپراقتىن يەنە بىر توپ ياش بوغۇنلار يېتىشىپ
چىقتى. خۇش ئاۋاز قۇشلار ئۇچۇپ كېلىپ بۇ قىيىنلارغا ئوۋا سالدى. قىيىنزارلىق
ھەركۈنى كۈن پېتىپ ئۇپۇق شەپەق قىزىللىقىغا چۆمۈلگەندە، شاختىن – شاخقا قونۇپ
يۈرگەن قۇشلارنىڭ شوخ سايراشلىرى بىلەن جانلىنىپ كېتىدىغان بولىدى. قۇشلار
سايرىماقتا. ئۇلار ئۆزلىرىنىڭ يېقىملىق ئاۋازلىرى بىلەن ئىككى تۈپ قېرى قېيىننىڭ
كۆڭۈل سىرلىرىنى ئىزھارلىماقتا. ئىككى تۈپ قېيىننىڭ كۆڭلى شاد. چۈنكى، ئۇلارنىڭ
ئۆمرى بىكار كەتمىدى. ئۇلارنىڭ قوينىدىمۇ قۇش ئوۋىلىرى بار. قۇشلار بۇ يەرنى
ئەۋلادمۇ ئەۋلاد ماكان ئېتىپ توختىماي سايراپ تۇرىدۇ. شۇنىڭ بىلەن، ئۇ خەلق بىلەن
كېڭەش ئۆتكۈزۈپ: «بىز بۇ تارىخنى ئېنىقلاشتا قانداق خاتالاشقان بولساق، كەلگۈسى
ئەۋلادلىرىمىزمۇ شۇنداق خاتالىشىدۇ. شۇڭا، بىز 12 ئايۋە ئاسماننىڭ بۇرجىغا
ئاساسلىنىپ، ھەر بىر يىلغا ئات قويايلى، بۇنىڭدىن كېيىن يىل ھېسابى شۇ يىللارنىڭ
ئايلىنىشى بىلەن ھېسابلىنىدىغان بولسۇن» دەپتۇ. ئۇنىڭ دېگىنىنى ىەلىقمۇ قوللاپتۇ.
سۇدىن ئەڭ ئاۋۋال چاشقان ئۆتۈپتۇ، شۇڭا يىل بېشى «چاشقان يىلى» دەپ ئاتىلىپتۇ.
ئۇنىڭدىن كېيىن ئۆتكەنلەر تۆۋەندىكى تەرتىپ بويىچەيىل ئىسمى بولۇپ قاپتۇ: ئۇي
(كالا) يىلى، بارىس يىلى (يولۋاس يىلى)، توشقان يىلى، لەھەڭ يىلى (تىمساھ يىلى)،
يىلان يىلى، ئات يىلى، قوي يىلى، مايمۇن يىلى، توخۇ يىلى، ئىت يىلى، توڭگۇز يىلى.
بىراق پاتىمە: - قورقماڭلار، بىز قول تۇتۇشۇپ ماڭايلى، - دەپتۇ. ئۇلار مېڭىپتۇ،
مېڭىپتۇ. يېقىنلىشىپ قارىسا، بىر يېنىۋاتقان گۈلخان ىكەن. ئۇ چۈشتىن بۇرۇن سائەت
ئوندا، 21 - ئەسىردىكى ئەڭ يېڭى ئالەم كېمىسىگە ئولتۇرۇپ يولغا چىقماقچى ئىدى. ئۇ
ئاي شارىغا چىقىپ ئولتۇراقلاشقان تۇغقانلىرىنى يوقلاپ ئۆتۈشنىمۇ كۆڭلىگە پۈككەنىدى.
ئالەم كېمىسىنڭ ئۇچۇشىغا يەنە يېرىم سائەت بار ئىكەن، نېمە قىلسام بولار؟ ئادىلجان
بىردەم ئويلىنىپ قالدى، ئاندىن ئېكرانلىق تېلىغونىنى ئېلىپ ئاندىن ئاي شارىدىكى
مومىسى بىلەن سۆزلەشمەكچى بولدى. مومىسى تېلىغۇن ئېكرانىدا پەيدا بولدى. «ئەسسالامۇ
ئەلەيكۇم، موما، ياخشى تۇردىڭىزمۇ؟ موما، ئىككى سائەتتىن كېيىن يېتىپ بارىمەن» دېدى
ئادىلجان. مومىسى: ۋاي - ۋۇي، نېمانچە تېز! دەۋەتتى. ئەلۋەتتە، _ دېدى ئادىلجان، _
مېنىڭ ئولتۇرغىنىم يېڭى تىپتىكى ئالەم كېمىسى! موما، ئاڭلىشىمچە، ئاي شارىدىكى
تەتقىقاتچىلار ھاۋانى بۇلغىمايدىغان قۇياش ئېنىرگىيىسى پىكاپىنى ياساپتۇدەك. بۇ
پىكاپ بازاردا بارمۇ؟ - دەپ سورىدى ئادىلجان. ئالغۇڭ بارمىدى؟ _ دېدى مومىسى كۈلۈپ،
_ ھەممە سودا سارايلىرىدا بار! خوش، موما، سەل تۇرۇپ كۆرۈشەيلى! - دېدى ئادىلجان،
ئاندىن ئالىقان كومپىيوتېرىنى ئاچتى. ئۇ كومپىيوتېر ئارقىلىق ئاي شارىدىكى ئەڭ چوڭ
سودا مەركىزىگە كىردى. مەن قۇياش ئېنىرگىيىسى پىكاپىدىن بىرنى سېتىۋالماقچىدىم،
چۈشتىن كېيىن سائەت ئىككىدە ئالغىلى بارىمەن... قەدىرلىك ساياھەتچىلەر، ئالەم
كېمىمىز ئۇچماقچى، بىخەتەرلىك تاسمىسىنى باغلىۋېلىڭلار! ئالەم كېمىسى ۋىڭ - ڭ قىلىپ
بوشلۇققا كۆتۈرۈلدى. بۇ چاغدا ئادىلجان بىخەتەرلىك تاسمىسىنى تېخى باغلاپ
بولالمىغانىدى، شۇڭا ئورنىدىن چۈشۈپ كەتتى. ۋايجان! - دەپ توۋلىۋەتتى ئادىلجان. ئۇ
كۆزىنى ئېچىپ ئەتراپقا ھەيرانلىق بىلەن قارىدى، ئاندىن: چۈشۈمكەن - دە دېدى.
ئادەملەر غەلىتە چۈشلەرنى كۆرىدۇ، ئاجايىپ خىياللارنى قىلىدۇ، بالىلار تېخىمۇ
شۇنداق. نۇرغۇن كەشپىياتچىلارنىڭ بارلىققا كېلىشىگە ئەنە شۇنداق چۈش ۋە خىياللار،
سەۋەبچى بولغان. ئايغا ۋە مارسقا چىقىش كىشىلەرنڭ خىيالى ئىدى، مانا ھازىر راست
ئىشقا ئايلاندى. مەسىلىلەر ئۈستىدە كۆزىتىشنى، ئويلىنىشىنى، ئىزدىنىشنى ئادەتكە
ئايلاندۇرسىڭىز، سىزمۇ چۈش ۋە خىياللىرىڭىزنى ئەمەلىيەتكە ئايلاندۇرالايسىز. مايمۇن
بىر توپ نەشپۈت كۆچىتى تىكىپتۇ ۋە ھەر كۈنى سۇ قۇيۇپ ئوغۇتلاپ، تېزرەك نەشپۈت
يېيىشنى كۈتۈپتۇ. نەشپۇت كۆچىتى ئەمدى كۆكىرىۋاتقاندا، قارىياغاچقا قونۇۋالغان بىر
قاغا دەپتۇ: - قاق – قاق! مايمۇن ئاكا، مايمۇن ئاكا، نەشپۈت كۆچىتى تىككىنىڭ
نېمىسى؟
